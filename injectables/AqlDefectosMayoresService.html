import { Injectable } from '@angular/core';
import { HttpClient } from '@angular/common/http';
import { Observable } from 'rxjs';
import { AqlDefectosMayoresModel as aqlModel } from '../models/aql-defectos-mayores.model';
import { environment } from '../../environments/environment';

// URL del back
const baseURL = environment.baseURL;

/**
 * Servicio del CRUD de la tabla `AQL_defectos_mayores`.
 */
@Injectable({
  providedIn: 'root'
})
export class AqlDefectosMayoresService {

  /**
   * @constructor
   * @param {HttpClient} http : Servicio HttpClient
   */
  constructor(private http: HttpClient) { }

  /**
   * @method getAllAql Obtiene todos los registros de la tabla `AQL_defectos_mayores`
   * @returns {Observable<aqlModel[]>} Observable que contiene un array de objetos del modelo de 
   * `AQL_defectos_mayores`.
   */
  getAllAql(): Observable<aqlModel[]> {
    return this.http.get<aqlModel[]>(`${baseURL}/aql`, {withCredentials: true});
  }

  /**
   * @method getOneAql Obtiene un registro de la tabla `AQL_defectos_mayores` en base a IDaql.
   * @param {Number} IDaql ID del registro a obtener.
   * @returns {Observable<aqlModel>} Oservable con un objeto como el modelo de `AQL_defectos_mayores`.
   */
  getOneAql(IDaql: number): Observable<aqlModel> {
    return this.http.get<aqlModel>(`${baseURL}/aql/${IDaql}`, { withCredentials: true});
  }

  /**
   * @method createAql Crea un nuevo registro de la tabla `AQL_defectos_mayores`.
   * @param {any} data Información para crear el nuevo registro.
   * @returns {Observable<any>} Observable que puede contener cualquiero cosa enviado desde el back.
   */
  createAql(data: any): Observable<any> {
    return  this.http.post(`${baseURL}/add-aql`, data, {withCredentials: true})
  }

  /**
   * @method updateAql Actualiza un registro de la tabla `AQL_defectos_mayores` con la infomración
   * proporcionada.
   * @param {Number} IDaql ID del registro a actualizar.
   * @param {any} data Nueva infomración del registro.
   * @returns {Observable<any>} Observable con cualquer coa enviada desde el back.
   */
  updateAql(IDaql: number, data: any): Observable<any> {
    return this.http.put(`${baseURL}/update-aql/${IDaql}`, data, { withCredentials: true})
  }

  /**
   * @method deleteAql Elimina un registro de la tabla `AQL_defectos_mayores`.
   * @param {Number} IDaql ID del registro a eliminar.
   * @returns {Observable<any>} Observable con cualquier información enviada desde el back.
   */
  deleteAql(IDaql: number): Observable<any> {
    return this.http.delete(`${baseURL}/delete-aql/${IDaql}`, {withCredentials: true})
  }
  
}
