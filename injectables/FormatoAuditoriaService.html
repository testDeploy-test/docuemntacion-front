import { Injectable } from '@angular/core';
import { HttpClient } from '@angular/common/http';
import { Observable } from 'rxjs';
import { FormatoAuditoriaModel as auditModel } from '../models/formato-auditoria.model';
import { environment } from '../../environments/environment';

const baseURL = environment.baseURL;

/**
 * Servicio del CRUD de la tabla `formato_auditoria`.
 */
@Injectable({
  providedIn: 'root'
})
export class FormatoAuditoriaService {

  /**
   * @constructor
   * @param {HttpClient} http : Servicio HttpClient 
   */
  constructor(private http: HttpClient) { }

  /**
   * @method getAllAudits Obtiene todos los registros de `formato_auditoria`.
   * @returns {Observable<auditModel[]>} Observable con un array de objetos como el modelo de `formato_auditoria`.
   */
  public getAllAudits(): Observable<auditModel[]> {
    return this.http.get<auditModel[]>(`${baseURL}/auditorias`, { withCredentials: true })
  }

  /**
   * @method getOneAudit Obtiene un registro de `formato_auditoria`.
   * @param {Number} IDauditoria ID del registro a obtener.
   * @returns {Observable<auditModel>} Observable con un objeto como el modelo de `formato_auditoria`.
   */
  public getOneAudit(IDauditoria: number): Observable<auditModel> {
    return this.http.get<auditModel>(`${baseURL}/auditoria/${IDauditoria}`)
  }

  /**
   * @method createAudit Crea un nuevo registro de `formato_auditoria`.
   * @param {any} data Información del nuevo registro. 
   * @returns {Observable<any>} Observable con cualquier cosa que devuelva el back.
   */
  public createAudit(data: any): Observable<any> {
    return this.http.post(`${baseURL}/add-auditoria`, data, { withCredentials: true})
  }

  /**
   * @method updateAudit Actaliza un registro de `formato_auditoria`.
   * @param {Number} IDauditoria ID del registro a actualizar. 
   * @param {any} data Nueva información del registro.
   * @returns {Observable<any>} Observable con cualquer cosa que devuelva el back. 
   */
  public updateAudit(IDauditoria: number, data: any): Observable<any> {
    return this.http.put(`${baseURL}/update-auditoria/${IDauditoria}`, data, { withCredentials: true})
  }

  /**
   * @method deleteAudit Elimina un registro de `formato_auditoria`.
   * @param {Number} IDauditoria ID del registro a eliminar.
   * @returns {Observable<any>} Observable con cualquier cosa que devuelva el back.
   */
  public deleteAudit(IDauditoria: number): Observable<any> {
    return this.http.delete(`${baseURL}/delete-auditoria/${IDauditoria}`, { withCredentials: true })
  }
}
