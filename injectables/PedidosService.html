import { Injectable } from '@angular/core';
import { HttpClient } from '@angular/common/http';
import { Observable } from 'rxjs';
import { PedidosModel } from '../models/pedidos.model';
import { environment } from '../../environments/environment';

const baseURL = environment.baseURL;

/**
 * Servicio del CRUD de la tabla `pedidos`.
 */
@Injectable({
  providedIn: 'root'
})
export class PedidosService {

  /**
   * @constructor
   * @param {HttpClient} http : Servicio HttpClient.
   */
  constructor(private http: HttpClient) { }

  /**
   * @method getAllOrders Obtiene todos los registros de `pedidos`.
   * @returns {Observable<PedidosModel[]>} Observable con un array de objetos como el modelo de `pedidos`.
   */
  public getAllOrders(): Observable<PedidosModel[]> {
    return this.http.get<PedidosModel[]>(`${baseURL}/pedidos`, { withCredentials: true})
  }

  /**
   * @method getOneOrder Obtiene un registro de `pedidos`.
   * @param codigoPedido Código del pedido a obtener.
   * @returns {Observable<PedidosModel>} Observable con un objeto como el modelo de `pedidos`.
   */
  public getOneOrder(codigoPedido: string): Observable<PedidosModel> {
    return this.http.get<PedidosModel>(`${baseURL}/pedido/${codigoPedido}`, { withCredentials: true })
  }

  /**
   * @method createOrder Crea un nuevo registro de `pedidos`.
   * @param {FormData} data Información del nuevo registro.
   * @returns {Observable<any>} Observable con cualquer cosa que devuelva el back.
   */
  public createOrder(data: FormData): Observable<any> {
    return this.http.post(`${baseURL}/add-pedido`, data, { withCredentials: true })
  }

  /**
   * @method updateOrder Actualiza un registro de `pedidos`.
   * @param {String} codigoPedido Código del pedido a actualizar.
   * @param {FormData} data Nueva información del registro.
   * @returns {Observable<any>} Observable con cualquier cosa que devuelva el back.
   */
  public updateOrder(codigoPedido: string, data: FormData): Observable<any> {
    return this.http.put(`${baseURL}/update-pedido/${codigoPedido}`, data, { withCredentials: true })
  }

  /**
   * @method deleteOrder Elimina un registro de `pedidos`.
   * @param {String} codigoPedido Código del pedido a eliminar.
   * @returns {Observable<any>} Observable con aualquier cosa que devuelva el back.
   */
  public deleteOrder(codigoPedido: string): Observable<any> {
    return this.http.delete(`${baseURL}/delete-pedido/${codigoPedido}`, { withCredentials: true })
  }
}
