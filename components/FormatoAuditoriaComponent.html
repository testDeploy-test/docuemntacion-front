import { Component, ElementRef, OnInit, Renderer2 } from '@angular/core';
import { Title } from '@angular/platform-browser';
import { FormatoAuditoriaService } from '../../../services/formato-auditoria.service';
import { FormatoAuditoriaModel } from '../../../models/formato-auditoria.model';
import { AuditoriaFormComponent } from '../../forms/auditoria-form/auditoria-form.component';
import { NoDataComponent } from '../../no-data/no-data.component';
import { AlertsService } from '../../../services/alerts.service';
import { SearchComponent } from '../../search/search.component';

/**
 * Componente de la tabla `formato_auditoria`
 * @implements {OnInit}
 */
@Component({
  selector: 'app-formato-auditoria',
  standalone: true,
  imports: [AuditoriaFormComponent, NoDataComponent, SearchComponent],
  templateUrl: './formato-auditoria.component.html',
  styleUrl: './formato-auditoria.component.css'
})
export class FormatoAuditoriaComponent implements OnInit {

  auditsList?: FormatoAuditoriaModel[]; // Declara la lista de registros de `formato_auditoria`

  auditFormVisble: boolean = false; // Visibilidad del formulario
  action: string = ""; // Acción del formulario ("create" o "update")
  selectedAudit: any; // Registro seleccionado para actualizar

  listLength: number = 0; // Cantidad de elementos totales de la tabla `formato_auditoria`
  table: string = "auditorias"; // Nombre de la tabla (ruta) para el componente search

  /**
   * @constructor
   * @param {Title} titleService Servicio del titulo
   * @param {FormatoAuditoriaService} auditsService Servicio de la tabla `formato_auditoria`
   * @param {AlertsService} alertsService Servicio de alertas
   * @param {ElementRef} el Servicio para acceder a elmentos del HTML
   * @param {Renderer2} renderer Servicio para la renderización del DOM
   */
  constructor(
    private titleService: Title, 
    private auditsService: FormatoAuditoriaService,
    private alertsService: AlertsService,
    private el: ElementRef,
    private renderer: Renderer2,
  ) {}

  /**
   * Al inicializar el componente cambia el titulo y obten los registros de `formato_auditoria`
   * @method
   */
  ngOnInit(): void {
    this.titleService.setTitle("Auditorias");
    this.getAll();
  }

  /**
   * Método para obtener todos los registros de la tabla `formato_auditoria`. Si es éxitosos asigna
   * los registros a `auditsList` y si falla muestra el error en la consola
   * @method
   */
  public getAll(): void {
    this.auditsService.getAllAudits().subscribe({
      next: (data) => {
        this.auditsList = data;
        this.listLength = data.length;
      },
      error: (e) => console.error(e)
    })
  }

  /**
   * Método para eliminar un registro de `formato_auditoria` en base a `IDauditoria`.
   * Pide confirmación de la acción. En caso de confirmar y si la acción es éxitosamuestra una 
   * alerta de éxito y recarga los registros, si falla muestra en consola el error y muestra una 
   * alerta de error
   * @method
   * @param IDauditoria 
   */
  public delete(IDauditoria: any): void {
    this.alertsService.confirmAlert("You are about to delete a record.", "delete").then((confirmed) => {
      if(confirmed) {
        this.auditsService.deleteAudit(IDauditoria).subscribe({
          next: () => {
            this.alertsService.succesAlert(`The record ${IDauditoria} deleted.`);
            this.getAll();
          },
          error: (e) => {
            console.error(e);
            this.alertsService.errorAlert();
          }
        })
      }
    })
  }

  /**
   * Método para visualizar el formulario con la acción de creación de un nuevo registro
   * @method
   */
  public showAuditFormCreate(): void {
    // Muestra el formulario
    this.auditFormVisble = true;
    // Especifica la acción
    this.action = "create";
    // Llama al método `disabled`
    this.disabled();
  }

  /**
   * Método para visualizar el formulario con la acción de actualización del registro
   * @method
   * @param audit 
   */
  public showAuditFormUpdate(audit: any): void {
    // Muestra el formulario
    this.auditFormVisble = true;
    // Asigna el valor de `selectedAudit`
    this.selectedAudit = audit;
    // Escpecifica la acción
    this.action = "update";
    // Llama al método `disabled`
    this.disabled()
  }

  /**
   * Método para deshabilitar todas las funciones de creación, actualización y 
   * eliminación.
   * @method
   * @returns {void} Retorna si no hay registros en la tabla `formato_auditoria`
   */
  public disabled(): void {
    if(!this.auditsList || this.auditsList.length === 0) return;
    const disabled = this.el.nativeElement.querySelector(".table-container");
    const button = this.el.nativeElement.querySelector(".form-create-button");

    this.renderer.addClass(disabled, "disabled");
    this.renderer.addClass(button, "disabled");
  }

    /**
   * Método para habilitar todas las funciones de creación, actualización y 
   * eliminación.
   * @method
   * @returns {void} Retorna si no hay registros en la tabla `formato_auditoria`
   */
  public enabled(): void {
    if(!this.auditsList || this.auditsList.length === 0) return;
    const disabled = this.el.nativeElement.querySelector(".table-container");
    const button = this.el.nativeElement.querySelector(".form-create-button");

    this.renderer.removeClass(disabled, "disabled");
    this.renderer.removeClass(button, "disabled");
  }

}

                                <button class="update-button" (click)="showAuditFormUpdate(audit)">Editar</button>
                                <button class="delete-button" (click)="delete(audit.IDauditoria)">Eliminar</button>
                            </div>
                        </th>
                    </tr>
                }
            </tbody>
        </table>
    </div>
}

@if(auditFormVisble) {
    <app-auditoria-form [action]="action" [selectedAudit]="selectedAudit" [auditsList]="auditsList" (enabledTable)="enabled()" (getAll)="getAll()" (hiddenForm)="auditFormVisble = $event"></app-auditoria-form>
}
