import { Component, ElementRef, OnInit, Renderer2 } from '@angular/core';
import { Title } from '@angular/platform-browser';
import { ItemsService } from '../../../services/items.service';
import { ItemsModel } from '../../../models/items.model';
import { NoDataComponent } from '../../no-data/no-data.component';
import { ItemsFormComponent } from '../../forms/items-form/items-form.component';
import { AlertsService } from '../../../services/alerts.service';
import { SearchComponent } from '../../search/search.component'; 

/**
 * Componente de la tabla `items`
 * @implements {OnInit}
 */
@Component({
  selector: 'app-items',
  standalone: true,
  imports: [ItemsFormComponent, NoDataComponent, SearchComponent],
  templateUrl: './items.component.html',
  styleUrl: './items.component.css'
})
export class ItemsComponent implements OnInit {

  itemsList?: ItemsModel[]; // Declara la lista de registros de `items`
  itemsFormVisible: boolean = false; // Visibilidad del formulario
  action: string = ""; // Acción del formulario ("create" o "update")
  selectedItem: any; // Registro seleccionado para actualizar

  listLength: number = 0; // Cantidadde elementos totales de la tabla `items`
  table: string = "items" // Nombre de la tabla (ruta) para el componente serch

  /**
   * @constructor
   * @param {Title} titleService Servicio del titulo
   * @param {ItemsService} itemsService Servicio de la tabla `items`
   * @param {AlertsService} alertsService Servicio de alertas
   * @param {ElementRef} el Servicio para acceder a elementos del HTML
   * @param {Renderer2} renderer Servicio para la renderización del DOM
   */
  constructor(
    private titleService: Title, 
    private itemsService: ItemsService,
    private alertsService: AlertsService,
    private el: ElementRef,
    private renderer: Renderer2,
  ) {}

  /**
   * Al inicializar el componente cambia el titulo y recupera los registros de `items`
   * @method
   */
  ngOnInit(): void {
    this.titleService.setTitle("Items");
    this.getAll();
  }

  /**
   * Método para obtener todos los registros de la tabla `items`. Si es éxitoso asigna los registros
   * a `itemsList` y si falla muestra el error en la consola
   * @method
   */
  public getAll(): void {
    this.itemsService.getAllitems().subscribe({
      next: (data) => {
        this.itemsList = data;
        this.listLength = data.length;
      },
      error: (e) => console.error(e)
    })
  }

  /**
   * Método para eliminar un registro de `items` en base a `codigoItem`.
   * Pide confirmación de la acción. En caso de confirmar y si la acción es éxitosa muestra una
   * alerta de éxito y recarga los registros, si falla muestra en consola el error y muestra una 
   * alerta de error
   * @method
   * @param codigoItem 
   */
  public delete(codigoItem: any): void {
    this.alertsService.confirmAlert("You are about to delete a record. <br> It is not recommended.", "delete").then((confirmed) => {
      if(confirmed) {
        this.itemsService.deleteItem(codigoItem).subscribe({
          next: () => {
            this.alertsService.succesAlert(`The record ${codigoItem} deleted.`);
            this.getAll();
          },
          error: (e) => {
            console.error(e);
            this.alertsService.errorAlert();
          }
        })
      }
    })
  }

  /**
   * Método para visualizar el formulario con la acción de creación de un nuevo registro
   * @method
   */
  public showItemFormCreate() : void {
    // Muestra el formulario
    this.itemsFormVisible = true;
    // Especifica la acción
    this.action = "create";
    // Llama al método `disabled`
    this.disabled();
  }

  /**
   * Método para visualizar el formulario con la acción de actualización del registro
   * @method
   * @param {any} item Toda la información del registroa a actualizar
   */
  public showItemFormUpdate(item: any): void {
    // Muestra el formulario
    this.itemsFormVisible = true;
    // Asigna el valor de `selectedItem`
    this.selectedItem = item;
    // Especifica la acción
    this.action = "update";
    // Llama al método `disabled`
    this.disabled();
  }

    /**
   * Método para deshabilitar todas las funciones de creación, actualización y 
   * eliminación.
   * @method
   * @returns {void} Retorna si no hay registros en la tabla `items`
   */
  public disabled() {
    if(!this.itemsList || this.itemsList.length === 0) return;
    const disabled = this.el.nativeElement.querySelector(".table-container");
    const button = this.el.nativeElement.querySelector(".form-create-button")
    this.renderer.addClass(disabled, "disabled");
    this.renderer.addClass(button, "disabled")
  }

  /**
   * Método para habilitar todas las funciones de creación, actualización y 
   * eliminación.
   * @method
   * @returns {void} Retorna si no hay registros en la tabla `items`
   */
  public enabled() {
    if(!this.itemsList || this.itemsList.length === 0) return;
    const disabled = this.el.nativeElement.querySelector(".table-container");
    const button = this.el.nativeElement.querySelector(".form-create-button")
    this.renderer.removeClass(disabled, "disabled");
    this.renderer.removeClass(button, "disabled")
  }

}

@if(!itemsList || itemsList.length === 0){
    <app-no-data (create)="showItemFormCreate()"></app-no-data>
} @else {
    <app-search [table]="table" [length]="itemsList.length" [listLength]="listLength" (all)="getAll()" (changeData)="itemsList = $event"></app-search>
    <button class="form-create-button" (click)="showItemFormCreate()">Crear nuevo</button>
    <div class="table-container">
        <table class="table">
            <thead class="table-header">
                <tr class="list-columns">
                    <th class="column">Código item</th>
                    <th class="column">Descripción</th>
                    <th class="column">Tecnologia</th>
                    <th class="column">Imagen</th>
                    <th class="column">Acciones</th>
                </tr>
            </thead>
            <tbody class="table-body">
                @for(item of itemsList; track item.codigoItem) {
                    <tr>
                        <td class="row">{{item.codigoItem}}</td>
                        <td class="row">{{item.descripcion}}</td>
                        <td class="row">{{item.tecnologia.nombre}}</td>
                        <td class="row"><img class="image" src="{{item.imagen}}" alt="{{item.codigoItem}}"></td>
                        <td class="row">
                            <div class="actions-container">
                                <button class="update-button" (click)="showItemFormUpdate(item)">Editar</button>
                                <button class="delete-button" (click)="delete(item.codigoItem)">Eliminar</button>
                            </div>
                        </td>
                    </tr>
                }
            </tbody>
        </table>
    </div>
}

@if(itemsFormVisible) {
    <app-items-form [action]="action" [selectedItem]="selectedItem" (enabledTable)="enabled()" (getAll)="getAll()" (hiddenForm)="itemsFormVisible = $event"></app-items-form>
}
