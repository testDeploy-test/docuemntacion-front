import { Component, EventEmitter, Input, Output } from '@angular/core';
import { SearchService } from '../../services/search.service';
import { provideIcons, NgIconComponent } from '@ng-icons/core';
import { bootstrapSearch } from "@ng-icons/bootstrap-icons";
import { FormsModule } from '@angular/forms';
import { AuditoriaDelProcesoService } from '../../services/auditoria-del-proceso.service';
import { PedidosService } from '../../services/pedidos.service';

/**
 * Componente de la barra de b√∫squeda
 */
@Component({
  selector: 'app-search',
  standalone: true,
  imports: [NgIconComponent, FormsModule],
  viewProviders: [provideIcons({bootstrapSearch})],
  templateUrl: './search.component.html',
  styleUrl: './search.component.css'
})
export class SearchComponent {

  /**
   * @constructor
   * @param {SearchService} search Servicio de b√∫squeda
   * @param {AuditoriaDelProcesoService} processService Servicio de `auditoria_del_proceso`
   * @param {PedidosService} ordersService Servicio de `pedidos`
   */
  constructor(
    private search: SearchService,
    private processService: AuditoriaDelProcesoService,
    private ordersService: PedidosService
  ) {}

  @Input() listLength: number = 0; // N√∫mero de elementos de la tabla original
  @Input() length: number = 0; // N√∫mero de elemtos de la tabla modificada (filtrada por la b√∫squeda)
  @Input() table: string = ""; // Nombre de la tabla a buscar
  
  @Output() changeData = new EventEmitter<any>; // Evento para modificar la informaci√≥n mostrada
  @Output() all = new EventEmitter<void>; // Evento para activar el m√©todo `getAll` del comonente padre

  query: string = ""; // Texto a buscar
  info: any; // Resultados de la b‚ñ∫3squeda

  /**
   * Emite el evento `all` 
   * @method
   */
  public get(): void {
    this.all.emit();
  }

  /**
   * M√©todo que activa la b√∫squeda de registros. Si el texto a buscar esta vac√≠o, activa el evento `all`
   * para mostrar todos los registros, en caso contrario activa el m√©todo `search` del servicio `search`
   * @method
   * @returns {void} Retorna para no seguir con la funci√≥n se se cumplen ciertos criterios.
   */
  public searchBy(): void {
    // Si `query` esta vacio activa el evento `all` y regresa.
    if(this.query === "") {
      if(this.length === this.listLength) return;
      this.all.emit()
      return;
    }
    // Si `query` no esta vac√≠o busca en la tabla especificada el texto proporcionado
    this.search.search(this.table, this.query).subscribe({
      next: (data) => {
        // Si es √©xitoso, pero no hay resultados y las tablas no son `auditoria_del_proceso` y `pedidos` retorna
        if(data.length === 0 && this.table !== "auditoria-proceso" && this.table !== "pedidos") return;
        // Si es √©xitoso y hay resultados, asigna el valor a `info`. (Tambi√©n se realiza si en las tablas anteriores no hay resultados)
        this.info = data;
        // Llama al m√©todo `searchUnionTables`
        this.searchUnionTables(data);
      },
      error: (e) => console.error(e)
    })
  }

  /**
   * Busca en la tabla uni√≥n de la tabla a buscar, activando los m√©todos correspondientes.
   * @method
   * @param {any} data 
   */
  searchUnionTables(data: any): void {
    // Si `data` esta vac√≠a convierte a `info` en un array vac√≠o
    if(data.length === 0) this.info = [];

    // Ejecuta una m√©todo dependiendo del caso
    switch(true) {
      case this.table === "auditoria-proceso":
        this.searchPD();
        break;
      case this.table === "pedidos":
        this.searchOI();
        break;
      default:
        this.changeData.emit(this.info);
        break;
    }
  }

  /**
   * Busca en `proceso_defecto` los registros que coincidan con el texto a buscar.
   * Si no existen registros que cumplan con el par√°metro retorna la ativaci√≥n del evento `all`.
   * Si existen registros agrega a `info` aquellos que no esten repetidos en base a la propiedad
   * `IDproceso` si hay registros en `info` activa el evento `changeData`
   * @method
   */
  searchPD(): void {
    // Busca en la tabla `proceso_defecto` (lo siguente es como viene en la ruta)
    this.search.search("procesos-defectos", this.query).subscribe({
      next: (list) => {
        // Si es √©xitoso...
        // Pero esta vac√≠o la respusta, activa el evento `changeData`
        if(list.length === 0) return this.changeData.emit(this.info);
        // Si no esta vac√≠o, itera en todos los elementos de la respuesta y busca un registro de `auditoria_del_proceso` en base a `IDproceso`
        for(let i = 0; i < list.length; i++) {
          this.processService.getOneProcess(list[i].IDproceso).subscribe({
            // Si es √©xitosa, agrega los elemntos con `IDproceso` diferente a `info`
            next: (results) => {
              if(!this.info.some((obj: any) => obj.IDproceso === results.IDproceso )) {
                this.info.push(results)
              }
              // Si es la √∫ltima iteraci√≥n e `info` sigue vac√≠o regresa, sino activa el evento `changeData`
              if(i === list.length - 1) {
                if(this.info.length === 0) return;
                this.changeData.emit(this.info)
              }
            },
            // Si falla imprime el error en consola
            error: (e) => console.error(e)
          })
        }
      },
      // Si falla imprime el error en consola
      error: (e) => console.error(e)
    })
  }

  /**
   * Busca en `pedido_item` los registros que coincidan con el texto a buscar.
   * Si no existen registros que cumplan con el par√°metro retorna la ativaci√≥n del evento `all`.
   * Si existen registros agrega a `info` aquellos que no esten repetidos en base a la propiedad
   * `codigoPedido` si hay registros en `info` activa el evento `changeData`
   * @method
   */
  searchOI(): void {
    // Busca en la tabla `pedido_item` (lo siguente es como viene en la ruta)
    this.search.search("pedidos-items", this.query).subscribe({
      // Si es √©xitoso...
      next: (list) => {
        // Pero la respuesta esta vac√≠a retorna la activaci√≥n del evento `changeData`
        if(list.length === 0) return this.changeData.emit(this.info);
        // Si no esta vac√≠o itera sobre caada elemento de la respuesta y busca un elemento de `pedidos` en base a `codigoPedido`
        for(let i = 0; i < list.length; i++) {
          this.ordersService.getOneOrder(list[i].codigoPedido).subscribe({
            // Si es √©xitosa agrega a `info` los elementos de `pedidos` que contengan un `codigoPedido` diferente a los existentes
            next: (results) => {
              if(!this.info.some((obj: any) => obj.codigoPedido === results.codigoPedido )) {
                this.info.push(results)
              }
              // Si es la √∫ltima iteraci√≥n e `info` esta vac√≠o regresa, sino activa el evento `changeData`
              if(i === list.length - 1) {
                if(this.info.length === 0) return;
                this.changeData.emit(this.info)
              }
            },
            // Si falla imprime el error en consola
            error: (e) => console.error(e)
          })
        }
      },
      // Si falla imrime el error en consola
      error: (e) => console.error(e)
    })
  }
}
