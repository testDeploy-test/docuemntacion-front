import { Component, OnInit, Input, Output, EventEmitter } from '@angular/core';
import { FormatoAuditoriaModel } from '../../../models/formato-auditoria.model';
import { FormatoAuditoriaService } from '../../../services/formato-auditoria.service';
import { Chart, ChartType } from 'chart.js/auto';
import { Subscription } from 'rxjs';
import { FilterService } from '../../../services/filter.service';

/**
 * Componente de la gráfica de Proporción de Auditorias con cada tipo de inspección de AQL
 * @implements {OnInit}
 */
@Component({
  selector: 'app-aql-graphics',
  standalone: true,
  imports: [],
  templateUrl: './aql-graphics.component.html',
  styleUrl: './aql-graphics.component.css'
})
export class AqlGraphicsComponent implements OnInit{

  /**
   * Al inicializar el componente obtiene los registros de `formato_auditoria` y deja la información de esta a 
   * disposición del método `createData`
   * @method
   */
  ngOnInit(): void {
    this.getAqlOfAudits().add(() => {
      this.createData();
    })
  }

  /**
   * @constructor
   * @param {FormatoAuditoriaService} auditService Servicio de la tabla `formato_auditoria`
   * @param {FilterService} filterService Servicio del filtro
   */
  constructor(private auditService: FormatoAuditoriaService, private filterService: FilterService) {
    this.filterService.$emitter.subscribe(() => {
      this.filter()
    })
  }

  @Input() fecha1: string = ""; // Parámetro enviado desde el padre. Utilizado para el filtrado (inicio)
  @Input() fecha2: string = ""; // Parámetro enviado desde el padre. Utilizado para el filtrado (final)

  @Output() chartVisible = new EventEmitter<boolean> // Evento para cambiar la visibilidad del componente

  chart: any; // Declara el gráfico
  list?: FormatoAuditoriaModel[]; // Declara la lista de registros de `formato_auditoria` (modificable)
  listOriginal?: FormatoAuditoriaModel[]; // Declara la lista de registros de `formato_auditoria` 

  sampleList: string[] = []; // Declara la lista para los tipos de inspección
  repeatList: number[] = []; // Declara la lista de las repeticiones de cada tipo de inspección

  /**
   * Método para obtener todos los registros de `formato_auditoria` ya que es la tabla necesaria para este
   * gráfico
   * @method
   * @returns {Subscription} Subscripción a los resultados de éste método para su futura utilización por
   * el método `createData`
   */
  getAqlOfAudits(): Subscription {
    return this.auditService.getAllAudits().subscribe({
      next: (data) => {
        this.list = data;
        this.listOriginal = data;
      },
      error: (e) => console.error(e)
    })
  }

  /**
   * Método para obtener la información necesaria para generar la gráfica.
   * Si `list` existe alamacena los tipos de inspección en `sampleList` y cuenta cuantas veces se repite y posteriormente
   * almacenalo en `repeatList`
   * @method
   * @returns {void} Si `list` existe regresa una vez completada toda la operación para no ocultar el 
   * gráfico
   */
  createData(): void {
    const aqlRepetitions = new Map<string, number>();
    
    if(this.list) {
      for (const dato of this.list) {
        const aql = dato.tipo_muestra;
        const insert = aql ? aql : "";
        if(aqlRepetitions.has(insert)) {
          aqlRepetitions.set(insert, aqlRepetitions.get(insert)! + 1);
        } else {
          aqlRepetitions.set(insert, 1)
        }
      }
      aqlRepetitions.forEach((repetitions, sample) => {
        this.sampleList.push(sample)
        this.repeatList.push(repetitions)
      })
      this.createAqlChart();
      return;
    }
    this.chartVisible.emit(false)
  }
  
  /**
   * Método para la creación del gráfico correspondiente.
   * Se especifica el tipo, y la información correspondiente extraida con el método `createData` además
   * de que se establecen unas configuraciónes del gráfico
   * @method
   */
  createAqlChart() {
    this.chart = new Chart("AqlChart", {
      type: "pie" as ChartType,
      data: {
        labels: this.sampleList,
        datasets: [ {
          data: this.repeatList
        }]
      },
      options: {
        responsive: true,
        plugins: {
          legend: { position: "left" }, 
          title: { display: true, text: "AQL Defectos Mayores"}
        },
        maintainAspectRatio: false
      }
    })
  }

  /**
   * Método para filtrar los registros por las fechas proporcionadas por el componente padre
   * `fecha1` y `fecha2`
   * @method
   * @returns {void} Regresa si no hay registros coincidentes con los parámetros establecidos u retorna
   * la ejecución del método `getAqlOfAudits` y subscribe a esto el método `createData` si los parámetros están
   * vacíos
   */
  filter() {
    // Destruye el gráfico actual
    this.chart.destroy();
    // Vacía las listas de los datos
    this.repeatList = [];
    this.sampleList = [];
    // Haz visible el gráfico
    this.chartVisible.emit(true)
    // Si los parámetros de filtrado están vacíos retorna la ejecución de `getAqlOfAudits` subscribiendo a esto a `createData` 
    if(this.fecha1 === "" && this.fecha2 === "" ) {
      return this.getAqlOfAudits().add(() => {
        this.createData();
      });
    };
    // Filtra la información según los parámetros
    const data = this.filterService.filterData(this.listOriginal, {fecha1: this.fecha1, fecha2: this.fecha2})
    // Si no hay registros que cumplan con los parámetros, oculata el gráfico y regresa
    if(data.length === 0) {
      this.chartVisible.emit(false)
      return;
    }
    // Si existen registros que cumplan los parámetros asignalos a `list` y vuleve a generar la información pertinente
    this.list = data;
    this.createData();
  }
}
