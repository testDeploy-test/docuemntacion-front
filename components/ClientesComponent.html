import { Component, ElementRef, OnInit, Renderer2 } from '@angular/core';
import { Title } from '@angular/platform-browser';
import { ClientesService } from '../../../services/clientes.service';
import { ClientesModel } from '../../../models/clientes.model';
import { NoDataComponent } from '../../no-data/no-data.component';
import { AlertsService } from '../../../services/alerts.service';
import { ClientesFormComponent } from '../../forms/clientes-form/clientes-form.component';
import { SearchComponent } from '../../search/search.component';

/**
 * Componente de la tabla `clientes`
 * @implements {OnInit}
 */
@Component({
  selector: 'app-clientes',
  standalone: true,
  imports: [ClientesFormComponent, NoDataComponent, SearchComponent],
  templateUrl: './clientes.component.html',
  styleUrl: './clientes.component.css'
})
export class ClientesComponent implements OnInit {

  customersList?: ClientesModel[]; // Declara la lista de registros de `clientes`
  public customersFormVisible: boolean = false; // Visibilidad del formulario
  action: string = ""; // Acción del formulario ("create" o "update")
  selectedCustomer: any; // Registro seleccionado para actualizar

  listLength: number = 0; // Cantidad de elementos totales de la tabla `clientes`
  table: string = "clientes" // Nombre de la tabla (ruta) para el componente search

  /**
   * @constructor
   * @param {Title} titleService 
   * @param {ClientesService} customersService 
   * @param {AlertsService} alertsService 
   * @param {ElementRef} el 
   * @param {Renderer2} renderer 
   */
  constructor(
    private titleService: Title, 
    private customersService: ClientesService,
    private alertsService: AlertsService,
    private el: ElementRef,
    private renderer: Renderer2,
  ) {}

  /**
   * Al inicializar el componente cambia el titulo y obtiene los registros de la tabla `pedidos`
   * @method
   */
  ngOnInit(): void {
    this.titleService.setTitle("Clientes");
    this.getAll();
  }

  /**
   * Método para obtener todos los registros de la tabla `pedidos`. Si es éxitoso asigna 
   * los registros a customersList y si falla muestra el error en la consola
   * @method
   */
  public getAll(): void {
    this.customersService.getAllCustomers().subscribe({
      next: (data) => {
        this.customersList = data;
        this.listLength = data.length;
      },
      error: (e) => console.error(e)
    })
  }

  /**
   * Método para eliminar un registro de `clientes` en base a `codigoCliente`.
   * Pide confirmación de la acción. En caso de confirmación y si la acción es éxitosa muestra una 
   * alerta de éxito y recarga los registros, si falla muestra el error en consola y muestra una 
   * alerta de error
   * @param {any} codigoCliente Código del registro a eliminar
   */
  public delete(codigoCliente: any): void {
    this.alertsService.confirmAlert("You are about to delete a record. <br> It is not recommended.", "delete").then((confirmed) => {
      if(confirmed) {
        this.customersService.deleteCustomer(codigoCliente).subscribe({
          next: () => {
            this.alertsService.succesAlert(`The record ${codigoCliente} deleted.`);
            this.getAll();
          },
          error: (e) => {
            console.error(e);
            this.alertsService.errorAlert();
          }
        })
      }
    })
  }


  /**
   * Método para visualizar el formualrio con la acción de creación de un nuevo registro.
   * @method
   */
  public showCutomerFormCreate(): void {
    // Muestra el formulario
    this.customersFormVisible = true;
    // Escpecifica la acción
    this.action = "create";
    // Llama al método `disabled`
    this.disabled();
  }

  /**
   * Método para la visualización del formulario con la acción de actualización del registro
   * @method
   * @param {any} customer Toda la información del registro a actualizar
   */
  public showCustomerFormUpdate(customer: any): void {
    // Muestra el formulario
    this.customersFormVisible = true;
    // Asigna el valor de `selectedCustomer`
    this.selectedCustomer = customer;
    // Especifica la acción
    this.action = "update";
    // Llama al método `dissabled`
    this.disabled();
  }

  /**
   * Método para deshabilitar todas las funciones de creación, actualización y 
   * eliminación.
   * @method
   * @returns {void} Retorna si no hay registros en la tabla `clientes`
   */
  public disabled() {
    if(!this.customersList || this.customersList.length === 0) return;
    const disabled = this.el.nativeElement.querySelector(".table-container");
    const button = this.el.nativeElement.querySelector(".form-create-button");
    this.renderer.addClass(disabled, "disabled");
    this.renderer.addClass(button, "disabled");
  }

  /**
   * Método para habilitar todas las funciones de creación, actualización y 
   * eliminación.
   * @method
   * @returns {void} Retorna si no hay registros en la tabla `clientes`
   */
  public enabled() {
    if(!this.customersList || this.customersList.length === 0) return;
    const disabled = this.el.nativeElement.querySelector(".table-container");
    const button = this.el.nativeElement.querySelector(".form-create-button");
    this.renderer.removeClass(disabled, "disabled");
    this.renderer.removeClass(button, "disabled");
  }
}

@if(!customersList || customersList.length === 0) {
    <app-no-data (create)="showCutomerFormCreate()"></app-no-data>
} @else {
    <app-search [table]="table" [listLength]="listLength" [length]="customersList.length" (all)="getAll()" (changeData)="customersList = $event"></app-search>
    <button class="form-create-button" (click)="showCutomerFormCreate()">Crear nuevo</button>
    <div class="table-container">
        <table class="table">
            <thead class="table-header">
                <tr class="list-columns">
                    <th class="column">Codigo Cliente</th>
                    <th class="column nombre">Nombre</th>
                    <th class="column">Acciones</th>
                </tr>
            </thead>
            <tbody class="table-body">
                @for(customer of customersList; track customer.codigoCliente) {
                    <tr>
                        <td class="row">{{customer.codigoCliente}}</td>
                        <td class="row">{{customer.nombre}}</td>
                        <td class="row">
                            <div class="actions-container">
                                <button class="update-button" (click)="showCustomerFormUpdate(customer)">Editar</button>
                                <button class="delete-button" (click)="delete(customer.codigoCliente)">Eliminar</button>
                            </div>
                        </td>
                    </tr>
                }
            </tbody>
        </table>
    </div>
}

@if(customersFormVisible) {
    <app-clientes-form [action]="action" [selectedCustomer]="selectedCustomer" (enabledTable)="enabled()" (getAll)="getAll()" (hiddenForm)="customersFormVisible = $event"></app-clientes-form>
}
