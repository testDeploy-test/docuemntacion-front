import { Component, Input, Output, EventEmitter, OnInit } from '@angular/core';
import { ItemsService } from '../../../services/items.service';
import { TecnologiasService } from '../../../services/tecnologias.service';
import { TecnologiasModel } from '../../../models/tecnologias.model';
import { AlertsService } from '../../../services/alerts.service';
import { ReactiveFormsModule } from '@angular/forms';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import { provideIcons, NgIconComponent } from '@ng-icons/core';
import { heroXMark } from '@ng-icons/heroicons/outline';
import { Subscription } from 'rxjs';

/**
 * Componente del formulario de `items`
 * @implements {OnInit}
 */
@Component({
  selector: 'app-items-form',
  standalone: true,
  imports: [ReactiveFormsModule, NgIconComponent],
  viewProviders: [provideIcons({heroXMark})],
  templateUrl: './items-form.component.html',
  styleUrl: './items-form.component.css'
})
export class ItemsFormComponent implements OnInit {

  /**
   * Al inicializar el componente obten los regitros de tecnologias y muestra el
   * formulario según la acción especificada en el componente padre
   * @method
   */
  ngOnInit(): void {
    this.getTechnologies().add(() => {
      if(this.action === "update") {
        this.showFormUpdate();
      } else {
        this.showFormCreate()
      }
    });
  }

  @Input() selectedItem: any; // Recibe la información del registro a actualizar
  @Input() action: string = ""; // Recibe la acción del formulario

  @Output() getAll = new EventEmitter<void> // Evento para activar el método `getAll` del padre
  @Output() hiddenForm = new EventEmitter<boolean> // Evento para cambiar la visibilidad del formulario en el componente padre
  @Output() enabledTable = new EventEmitter<void> // Evento para activar el método `enabled` del padre

  itemForm: FormGroup; // Declara el formulario
  technologiesList?: TecnologiasModel[]; // Declara la lista de registros de `tecnologias`
  selectedFile: File | any; // Declara la imagen

  /**
   * @constructor
   * @param {FormBuilder} fb Servicio para crear el formulario
   * @param {ItemsService} itemsService Servicio de la tabla `items`
   * @param {AlertsService} alertsService Servicio de alertas
   * @param {TecnologiasService} tecnologiesService Servicio de la tabla `tecnologias`
   */
  constructor(
    private fb: FormBuilder,
    private itemsService: ItemsService,
    private alertsService: AlertsService,
    private tecnologiesService: TecnologiasService
  ) {
    // Define los campos del formulario
    this.itemForm = this.fb.group({
      codigoItem: ["", Validators.required],
      descripcion: ["", Validators.required],
      IDtecnologia: [0, Validators.required],
      file: [null, Validators.required]
    });
  }

  /**
   * Método para obtener todos los registros de `tecnologias`. Si es éxitoso asigna los registros a `technologiesList`
   * y si falla muestra el error en la consola
   * @method
   * @returns {Subscription} Suscripción a los datos recabados por este método para su futura 
   * utilización por el método `showFormUpdate` o `showFormCreate`
   */
  public getTechnologies(): Subscription {
    return this.tecnologiesService.getAllTechnologies().subscribe({
      next: (data) => {
        this.technologiesList = data;
      },
      error: (e) => console.error(e)
    });
  }

  /**
   * Método para insertar el valor del campo file del formulario
   * @method
   * @param {any} event Input del campo `imagen`
   */
  onFileSelected(event: any): void {
    this.selectedFile = event.target.files[0];
    this.itemForm.patchValue({
      file: this.selectedFile
    })
  }

  /**
   * Método para crear un nuevo registro de `items` en base a la información recabada en el formulario
   * y convertido a `FormData`.
   * Si es éxitosa muestra una alerta de éxito, activa el evento `getAll` y llama al método `hidden`
   * y si falla muestra el error en la consola y muestra una alerta de error 
   * @method
   * @returns {void} Si el formulario es inválido envia una alerta para que complete
   * el formulario y retorna
   */  
  public create(): void {
    if(this.itemForm.invalid) {
      this.alertsService.completeAlert();
      return;
    }
    const formData = new FormData();
    formData.append('codigoItem', this.itemForm.get('codigoItem')?.value);
    formData.append('descripcion', this.itemForm.get('descripcion')?.value);
    formData.append('IDtecnologia', (this.itemForm.get('IDtecnologia')?.value));
    formData.append('file', this.selectedFile, this.selectedFile.name);

    this.itemsService.createItem(formData).subscribe({
      next: () => {
        this.alertsService.succesAlert("New Items record created.");
        this.getAll.emit();
        this.hidden();
      },
      error: (e) => {
        console.error(e);
        this.alertsService.errorAlert();
      }
    })
  }
  
  /**
   * Método para actualizar un registro de `items` en base a `codigoItem` del registro seleccionado
   * y en la nueva información proporcionada en el formulario convertida a `FormData`.
   * Pide confirmación de la acción. En caso de confirmar y que la acción sea éxitosa muestra una alerta de éxito,
   * emite el evento `getAll` y llamada al método `hidden` y si falla muestra el error en la consola y lanza
   * una alerta de error
   * @method
   */
  public update(): void {
    this.alertsService.confirmAlert("You are about to update a record.", "update").then((confirmed) => {
      if(confirmed) {
        const codigoItem = this.selectedItem.codigoItem;

        const formData = new FormData();
        formData.append('codigoItem', this.itemForm.get('codigoItem')?.value);
        formData.append('descripcion', this.itemForm.get('descripcion')?.value);
        formData.append('IDtecnologia', (this.itemForm.get('IDtecnologia')?.value));
        if(this.selectedItem.imagen !== this.itemForm.get("file")?.value) {
          formData.append('file', this.selectedFile, this.selectedFile.name);
        }

        this.itemsService.updateItem(codigoItem, formData).subscribe({
          next: () => {
            this.alertsService.succesAlert(`Record ${codigoItem} updated.`);
            this.getAll.emit();
            this.hidden();
          },
          error: (e) => {
            console.error(e);
            this.alertsService.errorAlert();
          }
        })
      }
    })
  } 

  /**
   * Método que vacía los campos del formulariocuando se visualiza para crear un nuevo registro
   * @method
   */  
  public showFormCreate(): void {
    this.itemForm.setValue({
      codigoItem: null,
      descripcion: null,
      IDtecnologia: null,
      file: null
    })
  }

  /**
   * Método para establecer el valor del registro a actualizar en los campos correspondientes del formulario
   * @method
   */  
  public showFormUpdate(): void {
    const id = this.technologiesList?.find(tech => tech.nombre === this.selectedItem.tecnologia.nombre)
    this.itemForm.setValue({
      codigoItem: this.selectedItem.codigoItem,
      descripcion: this.selectedItem.descripcion,
      IDtecnologia: id?.IDtecnologia,
      file: this.selectedItem.imagen
    })
  }

  /**
   * Método para cerrar el formulario detectando cambios en la información inicial tanto para la creación
   * de un nuevo registro o la actualización de un registro. En caso de detectar cambios pide una confirmación 
   * antes de cerrar el formulario en caso de confirmar llama al método `hidden`
   * @method
   * @returns Retorna si se cumplen ciertas condiciones para no ejecutar el resto de la lógica
   */  
  public close(): void {
    // Si la acción es para actualizar un registro...
    if(this.action === "update") {
      // Obtiene el `IDtecnologia` de la tecnologia utilizada en el item
      const id = this.technologiesList?.find(tech => tech.nombre === this.selectedItem.tecnologia.nombre)
      // Estructura inicial de la información del registro selleccionado para actualizar
      const dataPrev = this.selectedItem? {
        codigoItem: this.selectedItem.codigoItem,
        descripcion: this.selectedItem.descripcion,
        IDtecnologia: id?.IDtecnologia,
        file: this.selectedItem.imagen
      } : null

      // Si los datos previso y actuales son iguales llama al método `hidden` y regresa
      if(JSON.stringify(this.itemForm.value) === JSON.stringify(dataPrev)){
        this.hidden();
        return;
      }
      // Si los datos previos y actuales son diferentes pide confirmación para cerrar el formaulario
      this.alertsService.confirmAlert("The changes will be lost.", "close").then((confirmed) => {
        if(confirmed) {
          this.hidden();
          return;
        }
      })
      return;
    }

    // Si la acción es para crear un nuevo registro
    // Estructura inicial de la información
    const dataNew = {
      codigoItem: null,
      descripcion: null,
      IDtecnologia: null,
      file: null
    }

    // Si los datos previos y actuales son iguales llama al método `hidden` y regresa 
    if(JSON.stringify(this.itemForm.value) === JSON.stringify(dataNew)){
      this.hidden();
      return;
    }
    // Si los datos previos y actuales son diferentes pide confirmación para cerrar el cormulario 
    this.alertsService.confirmAlert("The data will be lost.", "close").then((confirmed) => {
      if(confirmed) {
        this.hidden();
        return;
      }
    })

  }

  /**
   * Método para ocultar el formulario y habilitar los botones de las acciónes en el componente padre
   * emitiendo los eventos `hiddenForm` y `enabledTable`
   * @method
   */    
  public hidden() {
    this.hiddenForm.emit(false);
    this.enabledTable.emit();
  }

}
