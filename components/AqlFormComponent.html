import { Component, Input, Output, EventEmitter,  OnInit } from '@angular/core';
import { ReactiveFormsModule } from '@angular/forms';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import { AqlDefectosMayoresService } from '../../../services/aql-defectos-mayores.service';
import { provideIcons, NgIconComponent } from '@ng-icons/core';
import { heroXMark } from '@ng-icons/heroicons/outline';
import { AlertsService } from '../../../services/alerts.service';

/**
 * Componente del formulario `AQL_defectos_mayores`
 * @implements {OnInit}
 */
@Component({
  selector: 'app-aql-form',
  standalone: true,
  imports: [ReactiveFormsModule, NgIconComponent],
  viewProviders: [provideIcons({heroXMark})],
  templateUrl: './aql-form.component.html',
  styleUrl: './aql-form.component.css'
})
export class AqlFormComponent implements OnInit {

  /**
   * Al inicializar el componente muestra el formulario según la acción especificada en el componente padre
   * @method
   */
  ngOnInit(): void {
    if(this.action === "update") {
      this.showFromUpdate()
    } else {
      this.showFormCreate()
    }
  }

  @Input() selectedAql: any; // Recibe la información del registro a actualizar
  @Input() action: string = ""; // Recibe la acción del formulario

  @Output() getAll = new EventEmitter<void> // Evento para activar el método `getAll` del padre
  @Output() hiddenForm = new EventEmitter<boolean> // Evento para cambiar la visibilidad del formulario en el padre
  @Output() enabledTable = new EventEmitter<void> // Evento para activar el método `enabled` del padre

  public aqlForm: FormGroup; // Declaración del formlario

  /**
   * @constructor
   * @param {FormBuilder} fb Servicio para crear formularios 
   * @param {AlertsService} alertsService Servicio de alertas
   * @param {AqlDefectosMayoresService} aqlService Servicio de la tabla `AQL_defectos_mayores`
   */
  constructor(
    private fb: FormBuilder,
    private alertsService: AlertsService,
    private aqlService: AqlDefectosMayoresService
  ) {
    // Declara los campos del formulario
    this.aqlForm = this.fb.group({
      minimo: ["", Validators.required],
      maximo: [""],
      muestra_inspeccion_reducida: ["", Validators.required],
      muestra_inspeccion_normal: ["", Validators.required],
      muestra_inspeccion_severa: ["", Validators.required]
    })
  }

  /**
   * Método para crear un nuevo registro de `AQL_defectos_mayores` en base a la información recabada en el formulario.
   * Si es éxitosa muestra una alerta de éxito, emite el evento `getAll` y oculta el formulario llamando al
   * método `hidden` y si falla muestra el error en la consola y muestra una alerta de error
   * @method
   * @returns {void} Retorna si el formulario es inválido para no ejecutar el resto de la lógica
   */
  public create(): void {
    if(this.aqlForm.invalid) {
      this.alertsService.completeAlert()
      return;
    }
    this.aqlService.createAql(this.aqlForm.value).subscribe({
      next: () => {
        this.alertsService.succesAlert("New AQL record created.")
        this.getAll.emit();
        this.hidden()
      },
      error: (e) => {
        console.error(e);
        this.alertsService.errorAlert()
      } 
    })

  }

  /**
   * Método para actualizar un registro de `AQL_defectos_mayores` en base al `IDaql` del registro seleccionado
   * y en la nueva información proporcionada en el formulario.
   * Pide confirmación de la acción. En caso de confirmar y que la acción sea éxitosa muestra una alerta de éxito,
   * emite el evento `getAll` y oculta el firmulario llamando al método `hidden` y si la acción falla muestra el error
   * en la consola y lanza una alerta de error
   * @method
   */
  public update(): void {
    this.alertsService.confirmAlert("You are about to update a record.", "update").then((confirmed) => {
      if(confirmed) {
        const IDaql = this.selectedAql.IDaql;
        this.aqlService.updateAql(IDaql, this.aqlForm.value).subscribe({
          next: () => {
            this.alertsService.succesAlert(`Record ${IDaql} updated`);
            this.getAll.emit();
            this.hidden()
          },
          error: (e) => {
            console.error(e);
            this.alertsService.errorAlert();
          }
        })
      }
    })
  }

  /**
   * Método que vacía los campos del formulario cuando se visualiza para crear un nuevo registro
   * @method
   */
  public showFormCreate(): void {
    this.aqlForm.setValue({
      minimo: null,
      maximo: null,
      muestra_inspeccion_reducida: null,
      muestra_inspeccion_normal: null,
      muestra_inspeccion_severa: null
    })
  }

  /**
   * Método para establecer el valor del registro a actualizar en sus respectivos campos en el formulario
   * @method
   */
  public showFromUpdate(): void {
    this.aqlForm.setValue({
      minimo: this.selectedAql.minimo,
      maximo: this.selectedAql.maximo,
      muestra_inspeccion_reducida: this.selectedAql.muestra_inspeccion_reducida,
      muestra_inspeccion_normal: this.selectedAql.muestra_inspeccion_normal,
      muestra_inspeccion_severa: this.selectedAql.muestra_inspeccion_severa
    })
  }

  /**
   * Método para cerrar el formulario detectando cambios en la información inicial tanto para la creación
   * de un nuevo registro o la actualización de un registro. En caso de detectar cambios pide una confirmación 
   * antes de cerrar el formulario en caso de confirmar llama al método `hidden`
   * @method
   * @returns Retorna si se cumplen ciertas condiciones para no ejecutar el resto de la lógica
   */
  public close() {
    // Estructura de la información inicial del formulario para crear un nuevo registro
    const dataNew = {
        minimo: null,
        maximo: null,
        muestra_inspeccion_reducida: null,
        muestra_inspeccion_normal: null,
        muestra_inspeccion_severa: null
    };

    // Estructura inicial de la información para actulizar un registro
    const dataPrev = this.selectedAql? {
        minimo: this.selectedAql.minimo,
        maximo: this.selectedAql.maximo,
        muestra_inspeccion_reducida: this.selectedAql.muestra_inspeccion_reducida,
        muestra_inspeccion_normal: this.selectedAql.muestra_inspeccion_normal,
        muestra_inspeccion_severa: this.selectedAql.muestra_inspeccion_severa
    } : null;

    // Si la acción es para actualizar un registro...
    if (this.action === 'update') {
      // Evalúa si los datos previso y los actuales son iguales de ser asi cierra el formulario.
        if (JSON.stringify(this.aqlForm.value) === JSON.stringify(dataPrev)) {
          this.hidden()
          return;
        }
        // Si los datos no son iguales pide confirmación para cerrar el formulario
        this.alertsService.confirmAlert('The changes will be lost', 'close').then((confirmed) => {
          if (confirmed) {
              this.hidden()
              return;
          }
      });
      return;
    }

    // Si los datos previos son iguales al del formulario cierra el formulario
    if (JSON.stringify(this.aqlForm.value) === JSON.stringify(dataNew)) {
      this.hidden()
      return;
    }
    // Si son distintos pide confirmación para cerrar el formulario
    this.alertsService.confirmAlert('The data will be lost.', 'close').then((confirmed) => {
      if (confirmed) {
          this.hidden()
          return;
      }
  });
  }

  /**
   * Método para ocultar el formulario y habilitar los botones de las acciónes en el componente padre
   * emitiendo los eventos `hiddenForm` y `enabledTable`
   * @method
   */
  public hidden() {
    this.hiddenForm.emit(false)
    this.enabledTable.emit()
  }

}
