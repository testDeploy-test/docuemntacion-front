import { Component, EventEmitter, Input, OnInit, Output } from '@angular/core';
import { ReactiveFormsModule } from '@angular/forms';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import { AlertsService } from '../../../services/alerts.service';
import { DefectosService } from '../../../services/defectos.service';
import { provideIcons, NgIconComponent } from '@ng-icons/core';
import { heroXMark } from '@ng-icons/heroicons/outline';

/**
 * Componente del formulario de `defectos`
 * @implements {OnInit}
 */
@Component({
  selector: 'app-defectos-form',
  standalone: true,
  imports: [ReactiveFormsModule, NgIconComponent],
  viewProviders: [provideIcons({heroXMark})],
  templateUrl: './defectos-form.component.html',
  styleUrl: './defectos-form.component.css'
})
export class DefectosFormComponent implements OnInit {

  /**
   * Al inicializar el componente muestra el formulario según la acción especificada en el componente padre
   * @method
   */
  ngOnInit(): void {
    if(this.action === "update") {
      this.showFormUpdate()
    } else {
      this.showFormCreate()
    }
  }

  @Input() selectedDefect: any; // Recibe la información del registro a actualizar
  @Input() action: string = ""; // Recibe la acción del formulario

  @Output() getAll = new EventEmitter<void> // Evento para activar el método `getAll` del padre
  @Output() hiddenForm = new EventEmitter<boolean> // Evento para cambiar la visibilidad del formulario en el padre
  @Output() enabledTable = new EventEmitter<void> // Evento para activar el método `enabled` del padre

  public defectsForm: FormGroup; // Declara el formulario

  /**
   * @constructor
   * @param {FormBuilder} fb Servicio para crear el formulario
   * @param {AlertsService} alertsService Servicio de alertas
   * @param {DefectosService} defectsService Servicio de la tabla `defectos`
   */
  constructor(
    private fb: FormBuilder,
    private alertsService: AlertsService,
    private defectsService: DefectosService,
  ) {
    // Define los campos ddel formulario
    this.defectsForm = this.fb.group({
      codigoDefecto: ["", Validators.required],
      nombre: ["", Validators.required],
      criterio_aceptacion: ["", Validators.required]
    })
  }

  /**
   * Método para crear un nuevo registro de `defectos` en base a la información recabada en el formulario.
   * Si es éxitosa muestra una alerta de éxito, activa el evento `getAll` y llama al método `hidden`
   * y si falla muestra el error en la consola y muestra una alerta de error 
   * @method
   * @returns {void} Si el formulario es inválido envia una alerta para que complete
   * el formulario y retorna
   */
  public create(): void {
    if(this.defectsForm.invalid){
      this.alertsService.completeAlert();
      return;
    }
    this.defectsService.createDefect(this.defectsForm.value).subscribe({
      next: () => {
        this.alertsService.succesAlert("New record of Defects created.");
        this.getAll.emit();
        this.hidden();
      },
      error: (e) => {
        console.error(e);
        this.alertsService.errorAlert();
        return;
      }
    })
  }

  /**
   * Método para actualizar un registro de `defectos` en base a `IDauditoria` del registro seleccionado
   * y en la nueva información proporcionada en el formulario.
   * Pide confirmación de la acción. En caso de confirmar y que la acción sea éxitosa muestra una alerta de éxito,
   * emite el evento `getAll` y llamada al método `hidden` y si falla muestra el error en la consola y lanza
   * una alerta de error
   * @method
   */  
  public update(): void {
    this.alertsService.confirmAlert("You are about to update a record.", "update").then((confirmed) => {
      if(confirmed) {
        const codigoDefecto = this.selectedDefect.codigoDefecto;
        this.defectsService.updateDefect(codigoDefecto, this.defectsForm.value).subscribe({
          next: () => {
            this.alertsService.succesAlert(`Record ${codigoDefecto} updated.`);
            this.getAll.emit();
            this.hidden();
          },
          error: (e) => {
            console.error(e);
            this.alertsService.errorAlert();
          }
        })
      }
    })
  }

  /**
   * Método que vacía los campos del formulariocuando se visualiza para crear un nuevo registro
   * @method
   */
  showFormCreate(): void {
    this.defectsForm.setValue({
      codigoDefecto: null,
      nombre: null,
      criterio_aceptacion: null
    });
  }

  /**
   * Método para establecer el valor del registro a actualizar en los campos correspondientes del formulario
   * @method
   */
  showFormUpdate(): void {
    this.defectsForm.setValue({
      codigoDefecto: this.selectedDefect.codigoDefecto,
      nombre: this.selectedDefect.nombre,
      criterio_aceptacion: this.selectedDefect.criterio_aceptacion
    });
  }

  /**
   * Método para cerrar el formulario detectando cambios en la información inicial tanto para la creación
   * de un nuevo registro o la actualización de un registro. En caso de detectar cambios pide una confirmación 
   * antes de cerrar el formulario en caso de confirmar llama al método `hidden`
   * @method
   * @returns Retorna si se cumplen ciertas condiciones para no ejecutar el resto de la lógica
   */  
  public close() {
    // Estructura inicial de la información de un nuevo registro
    const dataNew = {
      codigoDefecto: null,
      nombre: null,
      criterio_aceptacion: null
    };

    // Estructura inicial de la información del registro seleccionado para actualizar
    const dataPrev = this.selectedDefect? {
      codigoDefecto: this.selectedDefect.codigoDefecto,
      nombre: this.selectedDefect.nombre,
      criterio_aceptacion: this.selectedDefect.criterio_aceptacion
    } : null;

    // Si la acción es para actualizar un registro...
    if (this.action === 'update') {
      // Si los datos previos y actuales son iguales llama al método `hidden` y regresa
        if (JSON.stringify(this.defectsForm.value) === JSON.stringify(dataPrev)) {
          this.hidden()
          return;
        }
        // Si los datos previos y actuales son diferentes pide confirmación para cerrar el formulario
        this.alertsService.confirmAlert('The changes will be lost', 'close').then((confirmed) => {
          if (confirmed) {
              this.hidden()
              return;
          }
      });
      return;
    }

    // Si la acción es para agregar un registro...
    // Si los datos previos y actuales son iguales llama al método `hidden` y regresa
    if (JSON.stringify(this.defectsForm.value) === JSON.stringify(dataNew)) {
      this.hidden()
      return;
    }
    // Si los datos previso y actuales son diferentes pide confirmación para cerrar el formulario
    this.alertsService.confirmAlert('The data will be lost.', 'close').then((confirmed) => {
      if (confirmed) {
          this.hidden()
          return;
      }
  });
  }

  /**
   * Método para ocultar el formulario y habilitar los botones de las acciónes en el componente padre
   * emitiendo los eventos `hiddenForm` y `enabledTable`
   * @method
   */    
  public hidden() {
    this.hiddenForm.emit(false);
    this.enabledTable.emit();
  }
}
