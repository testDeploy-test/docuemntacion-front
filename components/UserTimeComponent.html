import { Component, Input, OnInit, Output, EventEmitter } from '@angular/core';
import { FormatoAuditoriaModel } from '../../../../models/formato-auditoria.model';
import { FormatoAuditoriaService } from '../../../../services/formato-auditoria.service';
import { Chart, ChartType } from "chart.js";
import { Subscription } from 'rxjs';
import { FilterService } from '../../../../services/filter.service';

/**
 * Componente de la gr√°fica `Tiempo dedicado a las auditorias`
 * @implements {OnInit}
 */
@Component({
  selector: 'app-user-time',
  standalone: true,
  imports: [],
  templateUrl: './user-time.component.html',
  styleUrl: './user-time.component.css'
})
export class UserTimeComponent implements OnInit {

  /**
   * Al inicializar el componente obtiene los registros de `formato_auditoria` y deja la informaci√≥n de esta 
   * a disposici√≥n del m√©todo `createUserTimeData`
   * @method
   */
  ngOnInit(): void {
    this.getAuditsList().add(() => {
      this.createUserTimeData()
    })
  }

  /**
   * @constructor
   * @param {FormatoAuditoriaService} auditService 
   * @param {FilterService} filterService 
   */
  constructor(private auditService: FormatoAuditoriaService, private filterService: FilterService) {
    this.filterService.$emitter.subscribe(() => {
      this.filter();
    })
  }

  @Input() fecha1: string = ""; // Par√°metro enviado desde el padre. Utilizado para el filtrado (inicio)
  @Input() fecha2: string = ""; // Par√°metro enviado desde el padre. Utilizado para el filtrado (final)

  @Output() chartVisible = new EventEmitter<boolean> // Evento para cambiar la visibilidad del componente

  chart: any; // Declara el gr√°fico

  list?: FormatoAuditoriaModel[]; // Declara la lista de registros de `formato_auditoria` (modificable)
  listOriginal?: FormatoAuditoriaModel[]; // Declara la lista de registros de `formato_auditoria`

  userList: string[] = []; // Declara la lista de los usuarios
  timeSummationList : number[] = []; // Declara la lista del total de tiempo dedicado por cada usuario a realizar auditorias
  averageTime: number[] = []; // Declara la lista del tiempo promedio dedicado a realizar auditorias de cada usuarios

  /**
   * M√©todo para obtener todos los registros de `formato_auditoria` ya que es la tabla necesaria para la
   * generaci√≥n de este gr√°fico
   * @method
   * @returns {Subscription} Subscripci√≥n a los resultados de √©ste m√©todo para su futura utilizaci√≥n por
   * el m√©todo `createUserTimeData`
   */
  getAuditsList(): Subscription {
    return this.auditService.getAllAudits().subscribe({
      next: (data) => {
        this.list = data;
        this.listOriginal = data;
      },
      error: (e) => console.error(e)   
    })
  }

  /**
   * M√©todo para obtener la informaci√≥n necesaria para generar el gr√°fico.
   * Si `list` existe almacena los usuarios en `userList` el tiempo promedio por auditoria en 
   * `averageTime` y el tiempo total dedicado de cada usuario en `timeSummationList` 
   * @method
   * @returns {void} Si `list` existe regresa una vez completada toda la operaci√≥n para no ocultar el 
   * gr√°fico
   */
  createUserTimeData(): void {
    if(this.list) {
      const userArray = Array.from(new Set(this.list.map(obj => obj.usuario.nombre)))
      const avgArray: number[] = []
      const summationTime: number[] = [];
      
      const userTime = new Map<string, number[]>();
    
      for(let i = 0; i < userArray.length; i++) {
        const timeOfUser = this.list.filter(obj => obj.usuario.nombre === userArray[i]).map(obj => obj.tiempo)
        const total = timeOfUser.reduce((acumulado, time) => acumulado + this.timeToMinutes(time), 0)
        summationTime.push(total)
        avgArray.push(total / timeOfUser.length)

        userTime.set(userArray[i], [avgArray[i], summationTime[i]])
      }

      const array = Array.from(userTime.entries()).sort((a, b) => b[1][1] - a[1][1]);

      array.forEach((value) => {
        this.userList.push(value[0])
        this.averageTime.push(value[1][0])
        this.timeSummationList.push(value[1][1])
      })
      this.createUserTimeChart();
      return;
    }
    this.chartVisible.emit(false);
  }

  /**
   * M√©todo para convertir el tiempo que tomo realizarse cada auditoria en minutos
   * @param {any} time Timepo requerido de cada auditoria
   * @returns {Number} Retorna la cantidad de tiempo en minutos
   */
  timeToMinutes(time: any) {
    const [hours, minutes, seconds] = time.split(':').map(Number);
    return (hours * 3600 + minutes * 60 + seconds)/60;
  }

  /**
   * M√©todo para la creaci√≥n del gr√°fico correspondiente.
   * Se especifica el tipo, y la informaci√≥n correspondiente extraida con el m√©todo `createUserTimeData` adem√°s
   * de que se establecen unas configuraci√≥nes del gr√°fico
   * @method
   */
  createUserTimeChart() {
    this.chart = new Chart("UserTimeChart", {
      data: {
        labels: this.userList,
        datasets: [{
          type: "bar" as ChartType,
          data: this.timeSummationList
        }, {
          label: "Tiempo promedio por auditoria",
          type: "line" as ChartType,
          data: this.averageTime,
          fill: false,
          backgroundColor: "#9AD0F5",
          borderColor: "#36A2EB"
        }]
      },
      options: {
        responsive: true,
        maintainAspectRatio: false,
        plugins: {
          title: { display: true, text: "Tiempo dedicado a las Auditorias"},
          legend: { display: false }
        },
        scales: {
          x: {
            title: { display: true, text: "Usuarios" }
          },
          y: { 
            title: { display: true, text: "Tiempo en minutos" }
           }
        }
      }
    })
  }

  /**
   * M√©todo para filtrar los registros por las fechas proporcionadas por el componente padre
   * `fecha1` y `fecha2`
   * @method
   * @returns {void} Regresa si no hay registros coincidentes con los par√°metros establecidos u retorna
   * la ejecuci√≥n del m√©todo `getAuditsList` y subscribe a esto el m√©todo `createUserTimeData` si los par√°metros est√°n
   * vac√≠os
   */  
  filter() {
    // Destruye el gr√°fico actual
    this.chart.destroy();
    // Vac√≠a las listas de datos
    this.userList = [];
    this.averageTime = [];
    this.timeSummationList = [];
    // Haz visible el gr√°fico
    this.chartVisible.emit(true)

    // Si los par√°metros de filtrado est√°n vac√≠os retorna la ejecuci√≥n de `getAuditsList` subscribiendo a esto a `createUserTimeData` 
    if(this.fecha1 === "" && this.fecha2 === "") {
      return this.getAuditsList().add(() => {
        this.createUserTimeData()
      })
    }

    // Filtra la informaci√≥n seg√∫n los par√°metros
    const data = this.filterService.filterData(this.listOriginal, { fecha1: this.fecha1, fecha2: this.fecha2 })
    // Si no hay registros que cumplan con los par√°metros, oculata el gr√°fico y regresa
    if(data.length === 0) {
      this.chartVisible.emit(false);
      return;
    }

    // Si existen registros que cumplan los par√°metros asignalos a `list` y vuleve a generar la informaci√≥n pertinente
    this.list =  data;
    this.createUserTimeData();
  }
}
