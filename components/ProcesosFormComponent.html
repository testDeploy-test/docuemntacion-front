import { Component, Input, Output, EventEmitter, OnInit } from '@angular/core';
import { AuditoriaDelProcesoModel } from '../../../models/auditoria-del-proceso.model';
import { AuditoriaDelProcesoService } from '../../../services/auditoria-del-proceso.service';
import { ProcesoDefectoModel } from '../../../models/proceso-defecto.model';
import { ProcesoDefectoService } from '../../../services/proceso-defecto.service';
import { FormArray, ReactiveFormsModule } from '@angular/forms';
import { FormBuilder, FormGroup, Validators} from '@angular/forms';
import { provideIcons, NgIconComponent } from '@ng-icons/core';
import { heroXMark } from '@ng-icons/heroicons/outline';
import { AlertsService } from '../../../services/alerts.service';
import { DefectosModel } from '../../../models/defectos.model';
import { DefectosService } from '../../../services/defectos.service';
import { FormatoAuditoriaModel } from '../../../models/formato-auditoria.model';
import { FormatoAuditoriaService } from '../../../services/formato-auditoria.service';
import { Subscription } from 'rxjs';

/**
 * Compoente del formulario de `auditoria_del_proceso`
 * @implements {OnInit}
 */
@Component({
  selector: 'app-procesos-form',
  standalone: true,
  imports: [ReactiveFormsModule, NgIconComponent],
  viewProviders: [provideIcons({heroXMark})],
  templateUrl: './procesos-form.component.html',
  styleUrl: './procesos-form.component.css'
})
export class ProcesosFormComponent implements OnInit {

  /**
   * Al inicializar el componente crea el formulario dinámico, obten la información de
   * `auditoria_del_proceso` y `proceso_defecto` y suscribe el resultado de estas a la vusalización del 
   * formulario según la acción especificada en el padre, además obtén los registros de 
   * `defectos` y `formato_auditoria`
   * @method
   */
  ngOnInit(): void {
    this.createDinamicForm();
    this.getProcess().add(() => {
      if(this.action === "update") {
        this.showFormUpdate();
      } else {
        this.showFormCreate();
      }
    });
    this.getPD().add(() => {
      if(this.action === "update") {
        this.showFormUpdate();
      } else {
        this.showFormCreate();
      }
    });
    this.getDeffects();
    this.getAudits();
    
  }

  @Input() selectedProcess: any; // Recibe la información del registro a actualizar
  @Input() action: string = ""; // Recibe la acción del formulario

  @Output() getAll = new EventEmitter<void> // Evento para activar el método `getAll` del padre
  @Output() getDP = new EventEmitter<void> // Evento para activar el método `getPD` del padre
  @Output() hiddenForm = new EventEmitter<boolean> // Evento para cambiar la visibilidad del formulario en el padre
  @Output() enabledTable = new EventEmitter<void> // Evento para activar el método `enabled` del padre

  selectedPD: any; // Declara la información de `proceso_defecto` cuand se quiera actualizar un registro de `auditoria_del_proceso`

  processList?: AuditoriaDelProcesoModel[]; // Declara la lista de registros de `auditoria_del_proceso`
  pdList?: ProcesoDefectoModel[]; // Declara la lista de registros de `proceso_defecto`
  deffectsList?: DefectosModel[]; // Declara la lista de registros de `defectos`
  auditsList?: FormatoAuditoriaModel[]; // Declara la lista de registros de `formato_auditoria`

  deffectsRemoved: any[] = []; // Delclara la lista de defectos removidos el formulario array

  processForm: FormGroup; // Declara el formulario principal
  pdForm: any; // Declara el formulario secundario
  pdFormArray: FormGroup // Declara el formulario array

  selectedFile: File | any; // Declara la imagen

  maxLenghtPDArray: number = 0; // Cantidad máxima de elemtos que puede tener el formulario array
  duplicated: boolean = false; // Para verificar si hay defectos repetidos en el proceso

  validAudit: boolean = true; // Para validar que la auditoria ingresada exista

  /**
   * @constructor
   * @param {AuditoriaDelProcesoService} processService Servicio de la tabla `auditoria_del_proceso`
   * @param {ProcesoDefectoService} pdService Servicio de la tabla `proceso_defecto`
   * @param {AlertsService} alertsService Servicio de alertas
   * @param {FormBuilder} fb Servicio para crear el formulario
   * @param {DefectosService} deffectsService Servicio de la tabla `defectos`
   * @param {FormatoAuditoriaService} auditsService Servico de la tabla `formato_auditoria`
   */
  constructor(
    private processService: AuditoriaDelProcesoService,
    private pdService: ProcesoDefectoService,
    private alertsService: AlertsService,
    private fb: FormBuilder,
    private deffectsService: DefectosService,
    private auditsService: FormatoAuditoriaService
  ) {
    // define los campos del formulario principal
    this.processForm = this.fb.group({
      pnc: ["", Validators.required],
      fecha_apertura: ["", Validators.required],
      cantidad_rechazada: [null],
      defectos_encontrados: ["", Validators.required],
      operario: ["", Validators.required],
      accion: ["", Validators.required],
      aprobado_por: [""],
      comentario_accion: [""],
      cantidad_descarte: [null],
      cantidad_aceptada: [null],
      observacion: [""],
      fecha_cierre: ["", Validators.required],
      re_auditoria: [""],
      estado_re_auditoria: [""],
      IDauditoria: [null, Validators.required],
      file: [null]
    });
    
    // Define los campos del formulario array
    this.pdFormArray = this.fb.group({
      data: this.fb.array([])
    })
  }

  /**
   * Método para inicializar el formulario array
   * @method
   */
  createDinamicForm(): void {
    this.pdFormArray;
  }

  /**
   * Facilita la obtención de la información del formulario array
   */
  get formArray(): FormArray {
    return this.pdFormArray.get("data") as FormArray
  }



  /**
   * Método para agregar un elemento `pdForm` al formulario array. Los campos son asignados
   * en función de la acción del formulario
   * @method
   */
  addPD(): void {
    let procesoId;
    if(this.action === "update") {
      procesoId = this.selectedProcess.IDproceso
      this.pdForm = this.fb.group({
        IDproceso_defecto: [""],
        IDproceso: [procesoId, Validators.required],
        codigoDefecto: ["", Validators.required]
      })
    } else {
      if(this.processList) {
        const last = this.processList?.at(-1)?.IDproceso
        procesoId = (last ? last : 0) + 1 
      }
      this.pdForm = this.fb.group({
        IDproceso: [procesoId, Validators.required],
        codigoDefecto: ["", Validators.required]
      })

    }

    this.formArray.push(this.pdForm);
  }

  /**
   * Método para eliminar un elemento del formulario array
   * @method
   * @param {Number} index index del elemento a eliminar
   * @returns {void} Retorna si el index es inválido
   */
  deletePD(index: number): void {
    if(index < 0 || index > this.formArray.length) {
      this.alertsService.errorAlert();
      return;
    }
    //no funciona de forma normal, encontre lo siguente para solucionarlo
    const value = this.formArray.value;

    this.formArray.setValue(
      value.slice(0, index).concat(
        value.slice(index + 1),
      ).concat(value[index]),
    );
    if(this.action === "update") {
      this.deffectsRemoved.push(this.formArray.at(value.length - 1).value)
    }
    this.hasDuplicated(null)
    this.formArray.removeAt(value.length - 1);
  }

  /**
   * Método para mantener la cantidad máxima definida en el formulario array.
   * Si los elementos son mas de los permitido elimina los sobrantes
   * @method
   * @param {any} event Input de `defectos_encontrados`
   */
  maxLenght(event: any): void {
    let max = event.target.value;
    if(max < this.maxLenghtPDArray) {
      while (this.maxLenghtPDArray > max) {
        this.formArray.removeAt(event.target.value)
        max++
      } 
    }
    this.maxLenghtPDArray = event.target.value;
  }

  /**
   * Método que verifica si hay elementos en el formulario array con el mismo `codigoDefecto`
   * @method
   */
  public hasDuplicated(event: any): void {
    let count: number = 0;
    this.duplicated = false; 
    if(event != null) {
      for (let i = 0; i < this.formArray.length; i++) {
        if (event.target.value === this.formArray.at(i).value.codigoDefecto) {
            count++;
            if (count >= 2) {
                this.duplicated = true;
            }
        }
    }
    }
    count = 0;
  }

  /**
   * Método para obtener todos los registros de `proceso_defecto` y si es éxitosa asignar la respuesta
   * a `pdList` y si falla mostrar el error en la consola
   * @method
   * @returns {Subscription} Suscripción de la infromación recabada en este método para su 
   * futura utilización en alguno de los métodos de visualización de formularios
   */
  public getPD(): Subscription {
    return this.pdService.getAllPD().subscribe({
      next: (data) => {
        this.pdList = data;
      },
      error: (e) => console.error(e)
    })
  }

  /**
   * Método para obtener todos los registros de `auditoria_del_proceso` y si es éxitosa asignar la respuesta
   * a `processList` y si falla mostrar el error en la consola
   * @method
   * @returns {Subscription} Suscripción de la infromación recabada en este método para su 
   * futura utilización en alguno de los métodos de visualización de formularios
   */
  getProcess(): Subscription {
    return this.processService.getAllProcesses().subscribe({
      next: (data) => {
        this.processList = data
      },
      error: (e) => console.error(e)
    })
  }

  /**
   * Método para obtener todos los registros de `defectos` y si es éxitosa asignar la respuesta
   * a `deffectsList` y si falla mostrar el error en la consola
   * @method
   */
  public getDeffects(): void {
    this.deffectsService.getAllDefects().subscribe({
      next: (data) => {
        this.deffectsList = data;
      },
      error: (e) => console.error(e)
    })
  }

  /**
   * Método para obtener todos los registros de `formato_auditoria` y si es éxitosa asignar la respuesta
   * a `auditsList` y si falla mostrar el error en la consola
   * @method
   */
  public getAudits(): void {
    this.auditsService.getAllAudits().subscribe({
      next: (data) => {
        this.auditsList = data;
      },
      error: (e) => console.error(e)
    })
  }

  /**
   * Método para asignar el valor del campo file a `processForm`
   * @method
   * @param {any} event Input imagen
   */
  onFileSelected(event: any): void {
    this.selectedFile = event.target.files[0];
    this.processForm.patchValue({
      file: this.selectedFile
    })
  }

  /**
   * Método para validar que la auditoria ingresa exista
   * @method
   * @param {any} event Input de IDauditoria
   */
  validateAudit(event: any): void {
    const value = event.target.value;
    this.validAudit = true;
    if(!this.auditsList?.some(audit => audit.IDauditoria === parseInt(value))) {
      this.validAudit = false;
    }
  }

  /**
   * Método que vacía los campos del formulariocuando se visualiza para crear un nuevo registro
   * @method
   */  
  showFormCreate(): void {
    this.processForm.setValue({
      pnc: null,
      fecha_apertura: null,
      cantidad_rechazada: null,
      defectos_encontrados: null,
      operario: null,
      accion: null,
      aprobado_por: null,
      comentario_accion: null,
      cantidad_descarte: null,
      cantidad_aceptada: null,
      observacion: null,
      fecha_cierre: null,
      re_auditoria: null,
      estado_re_auditoria: null,
      IDauditoria: null,
      file: null
    })
  }

  /**
   * Método para establecer el valor del registro a actualizar en los campos correspondientes de los formularios
   * @method
   */  
  public showFormUpdate(): void {

  /**
   * Método para actualizar un registro de `auditoria_del_proceso` en base a `IDproceso` del registro seleccionado
   * y en la nueva información proporcionada en el formulario.
   * Pide confirmación de la acción. En caso de confirmar y que la acción sea éxitosa llama al método `updatePD`,
   *  muestra una alerta de éxito, llamada a los métodos `hidden` y `getInfo` y si falla muestra el error en la 
   * consola y lanza una alerta de error
   * @method
   */
  public update(): void {
    this.alertsService.confirmAlert("You are about to update a record.", "update").then((confirmed) => {
      if(confirmed) {
        const IDproceso = this.selectedProcess.IDproceso;

        const formData = new FormData();
        formData.append("pnc", this.processForm.get("pnc")?.value)
        formData.append("fecha_apertura", this.processForm.get("fecha_apertura")?.value)
        formData.append("cantidad_rechazada", this.processForm.get("cantidad_rechazada")?.value)
        formData.append("defectos_encontrados", this.processForm.get("defectos_encontrados")?.value)
        formData.append("operario", this.processForm.get("operario")?.value)
        formData.append("accion", this.processForm.get("accion")?.value)
        formData.append("aprobado_por", this.processForm.get("aprobado_por")?.value)
        formData.append("comentario_accion", this.processForm.get("comentario_accion")?.value)
        formData.append("cantidad_descarte", this.processForm.get("cantidad_descarte")?.value)

        this.processService.updateProcess(IDproceso, formData).subscribe({

          }
        })

      }
    })
  }

  /**
   * Método para actualizar registros de `proceso_defecto` en base a `IDproceso_defecto`.
   * Si en la actualización del proceso se agregaron más defectos, crea los faltantes.
   * Si en la actualización del proceso Se cambiaron los registros existentes actualizalos uno por uno con la nueva información.
   * Si en la actualización del proceso se eliminaron registros eliminalos uno por uno
   * @method
   * @returns {void} Regresa si los registros de `proceso_defecto` y el formulario array son iguales
   */
  public updatePD(): void {
    // Si no hay cambios e la información regresa
    if(JSON.stringify(this.selectedPD) === JSON.stringify(this.formArray.value) && this.selectedProcess.defectos_encontrados === this.processForm.get("defectos_encontrados")?.value) return;
    
    // Si se eliminaron defectos, eliminalos
    // Elimina todos los defectos eliminados en el formulario debido al cambio de `defectos_encontrados`
    if(this.selectedProcess.defectos_encontrados > this.formArray.length) {
     const recordDeleted = this.selectedPD?.filter((pd: any) => {
      return !this.formArray.value.some((obj: any) => {
        return pd.IDproceso_defecto === obj.IDproceso_defecto
      })
     }).map((obj: any) => obj.IDproceso_defecto)

     for(let i = 0; i < recordDeleted.length; i++) {
      this.pdService.deletePD(recordDeleted[i]).subscribe({
        next: () => {
          console.log(`process - deffects ${recordDeleted[i]} removed.`);
        },
        error: (e) => {
          console.error(e);
          this.alertsService.errorAlert()
        }
      })
     }
    }
    if(this.deffectsRemoved.length !== 0) {
      for(let i = 0; i < this.deffectsRemoved.length; i++) {
        const IDproceso_defecto = this.deffectsRemoved[i].IDproceso_defecto;
        if(IDproceso_defecto !== undefined) {
          this.pdService.deletePD(IDproceso_defecto).subscribe({
            next: () => {
              console.log(`process - deffects ${IDproceso_defecto} removed.`);
            },
            error: (e) => {
              console.error(e);
              this.alertsService.errorAlert()
            }
          })
        }
      }
    }

    // Si se agregaron mas defectos crealos
    if(JSON.stringify(this.selectedPD) !== JSON.stringify(this.formArray.value)) {
      const data = this.formArray.value.filter((pd: any) => pd.IDproceso_defecto === "");
      data.map((obj: any) => {
        delete obj.IDproceso_defecto;
        return obj
      })
      this.pdService.createPD(data).subscribe({
        next: () => {
          console.log("new records created.");
        },
        error: (e) => {
          console.error(e);
          this.alertsService.errorAlert()
        }
      })
    }

    // Si se actualizaron los defectos existentes, actualizalos
    for(let i = 0; i < this.selectedPD.length; i++) {
      if(JSON.stringify(this.selectedPD[i]) !== JSON.stringify(this.formArray.at(i)?.value) && 
      this.selectedPD[i].IDproceso_defecto === this.formArray.get("IDproceso_defecto")?.value) {
        this.pdService.updatePD(this.selectedPD[i].IDproceso_defecto, this.formArray.at(i)?.value).subscribe({
          next: () => {
            console.log(`Record ${i + 1} updated.`)
          },
          error: (e) => {
            console.error(e);
            this.alertsService.errorAlert();
          }
        })
      }
    }

  }

  /**
   * Método para cerrar el formulario detectando cambios en la información inicial tanto para la creación
   * de un nuevo registro o la actualización de un registro. En caso de detectar cambios pide una confirmación 
   * antes de cerrar el formulario en caso de confirmar llama al método `hidden`
   * @method
   * @returns Retorna si se cumplen ciertas condiciones para no ejecutar el resto de la lógica
   */  
  public close(): void {
    // Si la acción a realizar es la actualización...
    if(this.action === "update") {
      // Estructura inicial de la información del registro a actualizar
      const processPrev = this.selectedProcess? {
        pnc: this.selectedProcess.pnc,
        fecha_apertura: this.selectedProcess.fecha_apertura,
        cantidad_rechazada: this.selectedProcess.cantidad_rechazada,
        defectos_encontrados: this.selectedProcess.defectos_encontrados,
        operario: this.selectedProcess.operario,
        accion: this.selectedProcess.accion,
        aprobado_por: this.selectedProcess.aprobado_por,
        comentario_accion: this.selectedProcess.comentario_accion,
        cantidad_descarte: this.selectedProcess.cantidad_descarte,
        cantidad_aceptada: this.selectedProcess.cantidad_aceptada,
        observacion: this.selectedProcess.observacion,
        fecha_cierre: this.selectedProcess.fecha_cierre,
        re_auditoria: this.selectedProcess.re_auditoria,
        estado_re_auditoria: this.selectedProcess.estado_re_auditoria,
        IDauditoria: this.selectedProcess.formato_auditorium.IDauditoria,
        file: this.selectedProcess.imagen
      } : null

      // Estructura inicial del formulario array del registro aa actualizar
      this.selectedPD = this.pdList?.filter(pd => pd.IDproceso === this.selectedProcess.IDproceso);
      const pdPrev = this.selectedPD?.map((pd: any) => ({
        IDproceso_defecto: pd.IDproceso_defecto,
        IDproceso: pd.IDproceso,
        codigoDefecto: pd.codigoDefecto
      }));

      // Si los datos previos y actual son iguales llama al método `hidden` y regresa
      if((JSON.stringify(this.processForm.value) === JSON.stringify(processPrev)) && (JSON.stringify(this.formArray.value) === JSON.stringify(pdPrev))) {
        this.hidden();
        return;
      }
      // Si los datos previos y actuales son diferentes pide confirmación para cerrar el formulario
      this.alertsService.confirmAlert("The changes will be lost.", "close").then((confirmed) => {
        if(confirmed) {
          this.hidden();
          return;
        }
      })
      return;
    }

    // Si la acción a realizar es la creación...
    // Estructura inicial de un nuevo registro
    const newProcess = {
        pnc: null,
        fecha_apertura: null,
        cantidad_rechazada: null,
        defectos_encontrados: null,
        operario: null,
        accion: null,
        aprobado_por: null,
        comentario_accion: null,
        cantidad_descarte: null,
        cantidad_aceptada: null,
        observacion: null,
        fecha_cierre: null,
        re_auditoria: null,
        estado_re_auditoria: null,
        IDauditoria: null,
        file: null
    }

    // Si los datos previos y actuales son iguales llama al método `hidden` y regresa
    if(JSON.stringify(this.processForm.value) === JSON.stringify(newProcess) && !this.formArray.at(0)) {
      this.hidden();
      return;
    }

    // Si los datos previos y actuales son diferentes pide confirmación para cerrar el formulario
    this.alertsService.confirmAlert("The data will be lost.", "close").then((confirmed) => {
      if(confirmed) {
        this.hidden();
        return;
      }
    })
  }
