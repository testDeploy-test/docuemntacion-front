import { Component, ElementRef, OnInit, Renderer2 } from '@angular/core';
import { Title } from '@angular/platform-browser';
import { AuditoriaDelProcesoService } from '../../../services/auditoria-del-proceso.service';
import { AuditoriaDelProcesoModel as processModel } from '../../../models/auditoria-del-proceso.model';
import { NoDataComponent } from '../../no-data/no-data.component';
import { AlertsService } from '../../../services/alerts.service';
import { ProcesoDefectoModel } from '../../../models/proceso-defecto.model';
import { ProcesoDefectoService } from '../../../services/proceso-defecto.service';
import { ProcesosFormComponent } from '../../forms/procesos-form/procesos-form.component';
import { SearchComponent } from '../../search/search.component';

/**
 * Componente de la tabla `audiotoria_del_proceso`
 * @implements {OnInit}
 */
@Component({
  selector: 'app-proceso',
  standalone: true,
  imports: [ProcesosFormComponent, NoDataComponent, SearchComponent],
  templateUrl: './auditoria-del-proceso.component.html',
  styleUrl: './auditoria-del-proceso.component.css'
})
export class AuditoriaDelProcesoComponent implements OnInit {

  /**
   * Al inicializar el componente recupera los registros de `auditoria_del_proceso` 
   * y `proceso_defecto` además de cambiar el titulo.
   * @method
   */
  ngOnInit(): void {
    this.getAll();
    this.titleService.setTitle("Auditoria del proceso");
    this.getPD()
  }

  processList?: processModel[]; // Declara la lista de registros de `auditoria_del_proceso`
  pdList?: ProcesoDefectoModel[]; // Declara la lista de los registros de `proceso_defecto`

  public processFormVisible: boolean = false; // Visibilidad del formulario de `auditoria_del_proceso`
  public action: string = ""; // Acción del formulario ("create" o "update")
  public selectedProcess: any; // Registro seleccionado para actualizar

  listLength: number = 0; // Cantidad de elementos totales de la tabla `auditoria_del_proceso`
  table: string = "auditoria-proceso"; // Nombre de la tabla (ruta) para el componente search

  /**
   * @constructor
   * @param {Title} titleService Servicio del titulo
   * @param {AuditoriaDelProcesoService} processService Servicio de la tabla `auditoria_del_proceso`
   * @param {AlertsService} alertsService Servicio de alertas
   * @param {ProcesoDefectoService} pdService Servicio de la tabla `proceso_defecto`
   * @param {ElementRef} el Servicio para acceder a elementos del HTML
   * @param {Renderer2} renderer Servicio para la renderización del DOM
   */
  constructor(
    private titleService: Title,
    private processService: AuditoriaDelProcesoService,
    private alertsService: AlertsService,
    private pdService: ProcesoDefectoService,
    private el: ElementRef,
    private renderer: Renderer2,
  ) {}


  /**
   * Método para obtener todos los registros de la tabla `auditoria_del_proceso`. Si es éxitoso
   * asigna los registros a `processList` y si falla muestra el error en la consola
   * @method
   */
  public getAll(): void {
    this.processService.getAllProcesses().subscribe({
      next: (data) => {
        this.processList = data;
        this.listLength = data.length;
      }, 
      error: (e) => console.error(e)
    })
  }

  /**
   * Método para obtener todos los registros de `proceso_defecto. Si es éxitoso asigna los registros
   * a `pdList` y si falla muestra el error en la consola.
   * @method
   */
  public getPD(): void {
    this.pdService.getAllPD().subscribe({
      next: (data) => {
        this.pdList = data;
      },
      error: (e) => console.error(e)
    })
  }

  /**
   * Método para eliminar un registro de `auditoria_del_proceso` en base a `IDproceso`.
   * Pide confirmación de la acción. En caso de confirmar y si la acción es éxitosa mustra una 
   * alerta de éxito y recarga los registros, si falla muestra en consola el error y muestra una
   * alerta de error.
   * @method
   * @param {any} IDproceso ID del registro a eliminar
   */
  public delete(IDproceso: any): void {
    this.alertsService.confirmAlert("You are about to delete a record.", "delete").then((confirmed) => {
      if(confirmed) {
        this.processService.deleteProcess(IDproceso).subscribe({
          next: () => {
            this.alertsService.succesAlert(`Record ${IDproceso} deleted.`);
            this.getAll()
          },
          error: (e) => {
            console.error(e);
            this.alertsService.errorAlert()
          }
        })
      }
    })
  }

  /**
   * Método para visulalizar el formulario con la acción de creación de un nuevo registro
   * @method 
   */
  public showCreateProcessForm(): void {
    // Muestra el formulario
    this.processFormVisible = true;
    // Especifica la acción
    this.action = "create";
    // Llama al método `disabled`
    this.disabled();
  }

  /**
   * Método para visualizar el formulario con la acción de actualización del registro.
   * @method
   * @param {any} process Toda la información del registro a actualizar
   */
  public showUpdateProcessForm(process: any): void {
    // Muestra el formulario
    this.processFormVisible = true;
    // Asigna el valor de `selectedProcess`
    this.selectedProcess = process;
    // Especifica la acción
    this.action = "update";
    // Llama al método `disabled`
    this.disabled();
  }
  

  /**
   * Método para deshabilitar todas las funciones de creación, actualización y 
   * eliminación.
   * @method
   * @returns {void} Retorna si no hay registros en la tabla `auditoria_del_proceso`
   */
  public disabled() {
    if(!this.processList || this.processList.length === 0) return;
    const disabled = this.el.nativeElement.querySelector(".table-container");
    const button = this.el.nativeElement.querySelector(".form-create-button");
    this.renderer.addClass(disabled, "disabled");
    this.renderer.addClass(button, "disabled");
  }

  /**
   * Método para habilitar todas las funciones de creación, actualización y 
   * eliminación.
   * @method
   * @returns {void} Retorna si no hay registros en la tabla `auditoria_del_proceso`
   */
  public enabled() {
    if(!this.processList || this.processList.length === 0) return;
    const disabled = this.el.nativeElement.querySelector(".table-container");
    const button = this.el.nativeElement.querySelector(".form-create-button");
    this.renderer.removeClass(disabled, "disabled");
    this.renderer.removeClass(button, "disabled");
  }

}

@if(processFormVisible) {
    <app-procesos-form [action]="action" [selectedProcess]="selectedProcess" (enabledTable)="enabled()" (getAll)="getAll()" (getDP)="getPD()" (hiddenForm)="processFormVisible = $event"></app-procesos-form>
}
