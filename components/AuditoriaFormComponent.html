import { Component, OnInit, Input, Output, EventEmitter, ElementRef } from '@angular/core';
import { FormatoAuditoriaService } from '../../../services/formato-auditoria.service';
import { AlertsService } from '../../../services/alerts.service';
import { AqlDefectosMayoresModel } from '../../../models/aql-defectos-mayores.model';
import { AqlDefectosMayoresService } from '../../../services/aql-defectos-mayores.service';
import { UsuariosModel } from '../../../models/usuarios.model';
import { UsuariosService } from '../../../services/usuarios.service';
import { PedidoItemModel } from '../../../models/pedido-item.model';
import { PedidoItemService } from '../../../services/pedido-item.service';
import { TecnologiasModel } from '../../../models/tecnologias.model';
import { TecnologiasService } from '../../../services/tecnologias.service';
import { ItemsModel } from '../../../models/items.model';
import { ItemsService } from '../../../services/items.service';
import { PedidosModel } from '../../../models/pedidos.model';
import { PedidosService } from '../../../services/pedidos.service';
import { ReactiveFormsModule } from '@angular/forms';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import { provideIcons, NgIconComponent } from '@ng-icons/core';
import { heroXMark } from '@ng-icons/heroicons/outline';

/**
 * Componente del formulario `formato_auditoria`
 * @implements {OnInit}
 */
@Component({
  selector: 'app-auditoria-form',
  standalone: true,
  imports: [ReactiveFormsModule, NgIconComponent],
  viewProviders: [provideIcons({heroXMark})],
  templateUrl: './auditoria-form.component.html',
  styleUrl: './auditoria-form.component.css'
})
export class AuditoriaFormComponent implements OnInit {

  /**
   * Al inicializar el componente muestra el formulario según la acción especificada en el componente padre y
   * obten la información necesaria para llenar el formulario
   * @method
   */
  ngOnInit(): void {
    if(this.action == "update") {
      this.showFormUpdate();
    } else {
      this.showFormCreate();
    }
    this.getAdditionalinformation()
  }

  @Input() selectedAudit: any; // Recibe la información del registro a actualizar
  @Input() action: any; // Recibe la acción del formulario
  @Input() auditsList: any; // Recibe la lista de registros de la tabla `formato_auditoria`

  @Output() getAll = new EventEmitter<void> // Evento para activar el método `getAll` del padre
  @Output() hiddenForm = new EventEmitter<boolean> // Evento para cambiar la visibilidad del formulario en el padre
  @Output() enabledTable = new EventEmitter<void> // Evento para activar el método `enabled` del padre

  public oiList?: PedidoItemModel[]; // Declara la lista de registros de `pedido_item`
  public usersList?: UsuariosModel[]; // Declara la lista de registros de `usuarios`
  public aqlList?: AqlDefectosMayoresModel[]; // Declara la lista de registros de `AQL_defectos_mayores`
  public technologiesList?: TecnologiasModel[]; //Ddeclara la lista de registros de `tecnologias`
  public ordersList?: PedidosModel[]; // Declara la lista de registros de `pedidos`
  public itemsList?: ItemsModel[]; // Declara la lista de registros de `items`

  auditForm: FormGroup; //Declara el formulario

  hora_inicio: string = ""; // Declara la hora en que se comenzó a realizar el formulario
  hora_fin: string = ""; // Declara la hora en que se envio el formulario

  duplicated: boolean = false; // Para verificar si hay registros duplicados
  validOrder: boolean = true; // Para verificar si el pedido existe

  /**
   * @constructor
   * @param {FormBuilder} fb Servicio para crear el formulario
   * @param {FormatoAuditoriaService} auditsService Servicio de la tabla `formato_auditoria`
   * @param {PedidoItemService} oiService Servicio de la tabla `pedido_item`
   * @param {UsuariosService} userService Servicio de la tabla `usuarios`
   * @param {AqlDefectosMayoresService} aqlService Servicio de la tabla `AQL_defectos_mayores`
   * @param {TecnologiasService} technologiesService Servicio de la tabla `tecnologias`
   * @param {PedidosService} ordersService Servicio de la tabla `pedidos`
   * @param {ItemsService} itemsService Servicio de la tabla `items`
   * @param {AlertsService} alertsService Servicio de alertas
   */
  constructor (
    private fb: FormBuilder,
    private auditsService: FormatoAuditoriaService,
    private oiService: PedidoItemService,
    private userService: UsuariosService,
    private aqlService: AqlDefectosMayoresService,
    private technologiesService: TecnologiasService,
    private ordersService: PedidosService,
    private itemsService: ItemsService,
    private alertsService: AlertsService,
  ) {
    // Define los campos del formulario
    this.auditForm = this.fb.group({
      PO: ["", Validators.required],
      codigoPedido: ["", Validators.required],
      codigoItem: ["", Validators.required],
      fecha_auditoria: ["", Validators.required],
      hora_inicio: ["", Validators.required],
      hora_fin: ["", Validators.required],
      IDusuario: [null, Validators.required],
      tipo_auditoria: ["", Validators.required],
      IDtecnologia: [null, Validators.required],
      alertas_proceso: [""],
      cantidad_lote: [null, Validators.required],
      tipo_muestra: ["", Validators.required],
      IDaql: [null, Validators.required],
      estado: ["", Validators.required]
    })
  }

  /**
   * Método para obtener todos los registros de `pedido_item`. Si es éxitoso
   * asigna los registros a `oiList` y si falla muestra el error en la consola
   * @method
   */
  public getOI(): void {
    this.oiService.getAllOI().subscribe({
      next: (data) => {
        this.oiList = data;
      },
      error: (e) => console.error(e)
    })
  }

  /**
   * Método para obtener todos los registros de `items`. Si es éxitoso
   * asigna los registros a `itemsList` y si falla muestra el error en la consola
   * @method
   */
  public getItems(): void {
    this.itemsService.getAllitems().subscribe({
      next: (data) => {
        this.itemsList = data;
      },
      error: (e) => console.error(e)
    })
  }

  /**
   * Método para obtener todos los registros de `tecnologias`. Si es éxitoso
   * asigna los registros a `technologiesList` y si falla muestra el error en la consola
   * @method
   */  
  public getTechnologies(): void {
    this.technologiesService.getAllTechnologies().subscribe({
      next: (data) => {
        this.technologiesList = data;
      },
      error: (e) => console.error(e)
    })
  }

  /**
   * Método para obtener todos los registros de `AQL_defectos_mayores`. Si es éxitoso
   * asigna los registros a `aqlList` y si falla muestra el error en la consola
   * @method
   */  
  public getAql(): void {
    this.aqlService.getAllAql().subscribe({
      next: (data) => {
        this.aqlList = data;
      },
      error: (e) => console.error(e)
    })
  }

  /**
   * Método para obtener todos los registros de `usuarios`. Si es éxitoso
   * asigna los registros a `usersList` y si falla muestra el error en la consola
   * @method
   */  
  public getUsers(): void {
    this.userService.getAllUsers().subscribe({
      next: (data) => {
        this.usersList = data;
      },
      error: (e) => console.error(e)
    })
  }

  /**
   * Método para obtener todos los registros de `pedidos`. Si es éxitoso
   * asigna los registros a `ordersList` y si falla muestra el error en la consola
   * @method
   */  
  public getOrders(): void {
    this.ordersService.getAllOrders().subscribe({
      next: (data) => {
        this.ordersList = data;
      },
      error: (e) => console.error(e)
    })
  }

  /**
   * Método para obtenr toda la información necesaria para el llenado del formulario.
   * Llama a múltiples métodos
   * @method
   */
  public getAdditionalinformation(): void {
    this.getUsers()
    this.getTechnologies()
    this.getAql()
    this.getOI()
    this.getOrders()
    this.getItems()
  }

  /**
   * Método para checar si hay registros duplicados de `codigoItem` en base a `codigoPedido`. En caso de haber un duplicado
   * establece `duplicated` como true.
   * @param {any} event Input del campo `codigoItem`
   * @returns {void} Regresa si al actualizar un registro y el valor de `codigoItem` es el mismo
   */
  hasDuplicated(event: any): void {
    this.duplicated = false;
    if(event != null) {
      for(let i = 0; i < this.auditsList.length; i++) {
        if(this.action === "update" && this.selectedAudit.IDauditoria === this.auditsList[i].IDauditoria && event.target.value === this.auditsList[i].codigoItem) return;
        if(event.target.value === this.auditsList[i].codigoItem && this.auditForm.get("codigoPedido")?.value === this.auditsList[i].codigoPedido) {
          this.duplicated = true;
        }
      }
    }
  }

  /**
   * Método para obtener la hora en que se abrió el formulario y asignar este valor a 
   * `hora_inicio`
   * @method
   */
  public addHoraInicio(): void {
    const fullDate = new Date();

    const hours = (fullDate.getHours()).toString().padStart(2,"0");
    const minutes = (fullDate.getMinutes()).toString().padStart(2,"0");
    const seconds = (fullDate.getSeconds()).toString().padStart(2, "0");

    this.hora_inicio = `${hours}:${minutes}:${seconds}`;
  }

  /**
   * Método para establecer los campos `hora_inicio`, `hora_fin`
   * y `fecha_auditoria` en el formulario
   * @method
   */
  public setTimeFields(): void {
    const fullDate = new Date();

    const hours = (fullDate.getHours()).toString().padStart(2,"0");
    const minutes = (fullDate.getMinutes()).toString().padStart(2,"0");
    const seconds = (fullDate.getSeconds()).toString().padStart(2, "0");

    const year = fullDate.getFullYear();
    const month = (fullDate.getMonth() + 1).toString().padStart(2, "0");
    const day = (fullDate.getDate()).toString().padStart(2, "0");

    this.hora_fin = `${hours}:${minutes}:${seconds}`;
    const date = `${year}-${month}-${day}`;


    this.auditForm.patchValue({
      hora_inicio: this.hora_inicio,
      hora_fin: this.hora_fin,
      fecha_auditoria: date
    })
  }

  /**
   * Método para verificar si el `codigoPedido` ingresado existe en `pedidos`.
   * Si éxiste establece `validOrder` como true y si no como false
   * @method
   */
  validateOrder(event: any): void {
    const value = event.target.value;
    this.validOrder = true;
    if(!this.ordersList?.some((order: any) => order.codigoPedido === value)) {
      this.validOrder = false;
    }
  }

  /**
   * Método que vacía los campos del formulariocuando se visualiza para crear un nuevo registro
   * @method
   */
  public showFormCreate(): void {
    this.addHoraInicio();
    this.auditForm.setValue({
      PO: null,
      codigoPedido: null,
      codigoItem: null,
      fecha_auditoria: null,
      hora_inicio: null,
      hora_fin: null,
      IDusuario: null,
      tipo_auditoria: null,
      IDtecnologia: null,
      alertas_proceso: null,
      cantidad_lote: null,
      tipo_muestra: null,
      IDaql: null,
      estado: null
    })
  }

  /**
   * Método para establecer el valor del registro a actualizar en los campos correspondientes del formulario
   * @method
   */
  public showFormUpdate(): void {
    this.auditForm.setValue({
      PO: this.selectedAudit.PO,
      codigoPedido: this.selectedAudit.codigoPedido,
      codigoItem: this.selectedAudit.codigoItem,
      fecha_auditoria: this.selectedAudit.fecha_auditoria,
      hora_inicio: this.selectedAudit.hora_inicio,
      hora_fin: this.selectedAudit.hora_fin,
      IDusuario: this.selectedAudit.usuario.IDusuario,
      tipo_auditoria: this.selectedAudit.tipo_auditoria,
      IDtecnologia: this.selectedAudit.tecnologia.IDtecnologia,
      alertas_proceso: this.selectedAudit.alertas_proceso,
      cantidad_lote: this.selectedAudit.cantidad_lote,
      tipo_muestra: this.selectedAudit.tipo_muestra,
      IDaql: this.selectedAudit.aql.IDaql,
      estado: this.selectedAudit.estado
    })
  }

  /**
   * Método para crear un nuevo registro de `formato_auditoria` en base a la información recabada en el formulario.
   * Si es éxitosa muestra una alerta de éxito, activa el evento `getAll` y llama al método `hidden`
   * y si falla muestra el error en la consola y muestra una alerta de error 
   * @method
   * @returns {void} Si el formulario es inválido envia una alerta para que complete
   * el formulario y retorna
   */
  public create(): void {
    this.setTimeFields();
    if(this.auditForm.invalid) {
      this.alertsService.completeAlert();
      return;
    }
    this.auditsService.createAudit(this.auditForm.value).subscribe({
      next: () => {
        this.alertsService.succesAlert("New audits record created.");
        this.getAll.emit();
        this.hidden();
      },
      error: (e) => {
        console.error(e);
        this.alertsService.errorAlert();
      }
    })
  }

  /**
   * Método para actualizar un registro de `formato_auditoria` en base a `IDauditoria` del registro seleccionado
   * y en la nueva información proporcionada en el formulario.
   * Pide confirmación de la acción. En caso de confirmar y que la acción sea éxitosa muestra una alerta de éxito,
   * emite el evento `getAll` y llamada al método `hidden` y si falla muestra el error en la consola y lanza
   * una alerta de error
   * @method
   */
  public update(): void {
    this.alertsService.confirmAlert("You are about to update a record.", "update").then((confirmed) => {
      if(confirmed) {
        const IDauditoria = this.selectedAudit.IDauditoria;
        this.auditsService.updateAudit(IDauditoria, this.auditForm.value).subscribe({
          next: () => {
            this.alertsService.succesAlert(`Record ${IDauditoria} updated.`);
            this.getAll.emit();
            this.hidden();
          },
          error: (e) => {
            console.error(e);
            this.alertsService.errorAlert();
          }
        })
      }
    })
  }

  /**
   * Método para cerrar el formulario detectando cambios en la información inicial tanto para la creación
   * de un nuevo registro o la actualización de un registro. En caso de detectar cambios pide una confirmación 
   * antes de cerrar el formulario en caso de confirmar llama al método `hidden`
   * @method
   * @returns Retorna si se cumplen ciertas condiciones para no ejecutar el resto de la lógica
   */
  public close(): void {
    // Si la acción es para actualizar un registro..
    if(this.action === "update") {
      // Estructura de la información del registro seleccionado
      const dataPrev = this.selectedAudit? {
        PO: this.selectedAudit.PO,
        codigoPedido: this.selectedAudit.codigoPedido,
        codigoItem: this.selectedAudit.codigoItem,
        fecha_auditoria: this.selectedAudit.fecha_auditoria,
        hora_inicio: this.selectedAudit.hora_inicio,
        hora_fin: this.selectedAudit.hora_fin,
        IDusuario: this.selectedAudit.usuario.IDusuario,
        tipo_auditoria: this.selectedAudit.tipo_auditoria,
        IDtecnologia: this.selectedAudit.tecnologia.IDtecnologia,
        alertas_proceso: this.selectedAudit.alertas_proceso,
        cantidad_lote: this.selectedAudit.cantidad_lote,
        tipo_muestra: this.selectedAudit.tipo_muestra,
        IDaql: this.selectedAudit.aql.IDaql,
        estado: this.selectedAudit.estado
      }: null

      // Si los datos previos y actuales son iguales llama al método `hidden` y retorna
      if(JSON.stringify(this.auditForm.value) === JSON.stringify(dataPrev)){
        this.hidden();
        return;
      }
      // Si la información previa y la actual son diferentes pide confirmación para llamar al método `hidden`
      this.alertsService.confirmAlert("The changes will be lost.", "close").then((confirmed) => {
        if(confirmed) {
          this.hidden();
          return;
        }
      })
      return;
    }
    // Si la acción es para crear un nuevo registro...
    // Estructura de la información inicial de un nuevo registro
    const newData = {
      PO: null,
      codigoPedido: null,
      codigoItem: null,
      fecha_auditoria: null,
      hora_inicio: null,
      hora_fin: null,
      IDusuario: null,
      tipo_auditoria: null,
      IDtecnologia: null,
      alertas_proceso: null,
      cantidad_lote: null,
      tipo_muestra: null,
      IDaql: null,
      estado: null
    }

    // Si no hay cambios llama al método `hidden` y regresa
    if(JSON.stringify(this.auditForm.value) === JSON.stringify(newData)){
      this.hidden();
      return;
    }
    // Si hay cambios pide confirmación para llamar al método `hidden`
    this.alertsService.confirmAlert("The data will be lost.", "close").then((confirmed) => {
      if(confirmed) {
        this.hidden();
        return
      }
    })
  }

  /**
   * Método para ocultar el formulario y habilitar los botones de las acciónes en el componente padre
   * emitiendo los eventos `hiddenForm` y `enabledTable`
   * @method
   */  
  public hidden() {
    this.hiddenForm.emit(false)
    this.enabledTable.emit()
  }
