import { Component, EventEmitter, Output } from '@angular/core';
import { provideIcons, NgIconComponent } from '@ng-icons/core';
import {heroExclamationCircle as exclamation} from "@ng-icons/heroicons/outline"

/**
 * Componente que se muestra cuando no hay registros en las tablas.
 */
@Component({
  selector: 'app-no-data',
  standalone: true,
  imports: [NgIconComponent],
  viewProviders: [provideIcons({exclamation})],
  templateUrl: './no-data.component.html',
  styleUrl: './no-data.component.css'
})
export class NoDataComponent {

  // Emite un evento que asigne el componete padre.
  @Output() create = new EventEmitter<void>

  /**
   * Emite el evento que se asigne desde el componete padre, en este caso muestra el formulario de 
   * creaci√≥n de un registro, segun en la tabla que se este visualizando, solo si no existen registros
   * en dicha tabla
   * @method
   */
  public FirstRecord(): void {
    this.create.emit()
  }

}
