import { Component, ElementRef, OnInit, Renderer2 } from '@angular/core';
import { Title } from '@angular/platform-browser';
import { TecnologiasService } from '../../../services/tecnologias.service';
import { TecnologiasModel } from '../../../models/tecnologias.model';
import { NoDataComponent } from '../../no-data/no-data.component';
import { TecnologiasFormComponent } from '../../forms/tecnologias-form/tecnologias-form.component';
import { AlertsService } from '../../../services/alerts.service';
import { SearchComponent } from '../../search/search.component';

/**
 * Componente de la tabla `tecnologias`
 * @implements {OnInit}
 */
@Component({
  selector: 'app-tecnologias',
  standalone: true,
  imports: [TecnologiasFormComponent, NoDataComponent, SearchComponent],
  templateUrl: './tecnologias.component.html',
  styleUrl: './tecnologias.component.css'
})
export class TecnologiasComponent implements OnInit {

  /**
   * Al inicializar el componente cambia el titulo y recupera los registros de la
   * tabla `tecnologias` mediante la llamada al método `getAll`
   * @method
   */
  ngOnInit(): void {
    this.titleService.setTitle("Tecnologias");
    this.getAll();
  }

  /**
   * @constructor
   * @param {Title} titleService Servicio del titulo
   * @param {TecnologiasService} technologiesService Servicio de la tabla `tecnologias`
   * @param {AlertsService} alertsService Servicio de alertas
   * @param {ElementRef} el Servicio para acceder a elementos del HTML
   * @param {Renderer2} renderer Servicio para la renderización del DOM
   */
  constructor(
    private titleService: Title, 
    private technologiesService: TecnologiasService,
    private alertsService: AlertsService,
    private el: ElementRef,
    private renderer: Renderer2,
  ) {}


  technologiesList?: TecnologiasModel[]; // Declara la lists de registros de `tecnologias`
  technologyFormVisible: boolean = false; // Visibilidad del formulario de `tecnologias`
  action: string = ""; // Acción del formulario ("create" o "update")
  selectedTechnology: any; // Registro seleccionado para actualizar

  listLength: number = 0; // Cantidad de elementos totales de la tabla `tecnologias`
  table: string = "tecnologias" // Nombre de la tabla (ruta) para el componente search

  /**
   * Método para obtener todos los registros de la tabla `tecnologias`. Si es éxitoso asigna
   * los registros a `technologiesList` y si falla muestra el error en consola
   * @method
   */
  public getAll(): void {
    this.technologiesService.getAllTechnologies().subscribe({
      next: (data) => {
        this.technologiesList = data;
        this.listLength = data.length;
      },
      error: (e) => console.error(e)
    })
  }

  /**
   * Método para eliminar un registro de `tecnologias` en base a `IDtecnologia`.
   * Pide confirmación de la acción. En caso de confirmar y si la ección es éxitosa muestra una
   * alerta de éxito y recarga los registros, si falla muestra en consola el error y muestra una
   * alerta de error
   * @method
   * @param {any} IDtecnologia ID del registro a eliminar
   */
  public delete(IDtecnologia: any): void {
    this.alertsService.confirmAlert("You are about to delete a record. <br> It is not recommended.", "delete").then((confirmed) => {
      if(confirmed) {
        this.technologiesService.deleteTechnology(IDtecnologia).subscribe({
          next: () => {
            this.alertsService.succesAlert(`The record ${IDtecnologia} was deleted.`);
            this.getAll();
          },
          error: (e) => {
            console.error(e);
            this.alertsService.errorAlert()
          }
        })
      }
    })
  }

  /**
   * Método para visualizar el formulario con la acción de creación de un nuevo registro
   * @method
   */
  public showTechnologyFormCreate() {
    // Muestra el formulario
    this.technologyFormVisible = true;
    // Especifica la acción
    this.action = "create";
    // Llama al método `disabled`
    this.disabled();
  }

  /**
   * Método para visualizar el formulario con la cción de actualización de un registro
   * @method
   * @param {any} technology Toda la información del registro a actualizar
   */
  public showTechnologyFormUpdate(technology: any) {
    // Muestra el formulario
    this.technologyFormVisible = true;
    // Asigna el valor de `selectedTechnology`
    this.selectedTechnology = technology;
    // Especifica la acción
    this.action = "update";
    // Llama al método `disabled`
    this.disabled();
  }

  /**
   * Método para deshabilitar todas las funciones de creación, actualización y 
   * eliminación.
   * @method
   * @returns {void} Retorna si no hay registros en la tabla `tecnologias`
   */
  public disabled() {
    if(!this.technologiesList || this.technologiesList.length === 0) return;
    const disabled = this.el.nativeElement.querySelector(".table-container");
    const button = this.el.nativeElement.querySelector(".form-create-button");
    this.renderer.addClass(disabled, "disabled");
    this.renderer.addClass(button, "disabled");
  }

  /**
   * Método para habilitar todas las funciones de creación, actualización y 
   * eliminación.
   * @method
   * @returns {void} Retorna si no hay registros en la tabla `tecnologias`
   */
  public enabled() {
    if(!this.technologiesList || this.technologiesList.length === 0) return;
    const disabled = this.el.nativeElement.querySelector(".table-container");
    const button = this.el.nativeElement.querySelector(".form-create-button");
    this.renderer.removeClass(disabled, "disabled");
    this.renderer.removeClass(button, "disabled");
  }
}

@if(!technologiesList || technologiesList.length === 0) {
    <app-no-data (create)="showTechnologyFormCreate()"></app-no-data>
} @else {
    <app-search [length]="technologiesList.length" [listLength]="listLength" [table]="table" (changeData)="technologiesList = $event" (all)="getAll()"></app-search>
    <button class="form-create-button" (click)="showTechnologyFormCreate()">Crear nuevo</button>
    <div class="table-container">
        <table class="table">
            <thead class="table-header">
                <tr class="list-columns">
                    <th class="column">ID tecnologia</th>
                    <th class="column nombre">Nombre</th>
                    <th class="column">Acciones</th>
                </tr>
            </thead>
            <tbody class="table-body">
                @for(technology of technologiesList; track technology.IDtecnologia) {
                    <tr>
                        <td class="row">{{technology.IDtecnologia}}</td>
                        <td class="row">{{technology.nombre}}</td>
                        <td class="row">
                            <div class="actions-container">
                                <button class="update-button" (click)="showTechnologyFormUpdate(technology)">Editar</button>
                                <button class="delete-button" (click)="delete(technology.IDtecnologia)">Eliminar</button>
                            </div>
                        </td>
                    </tr>
                }
            </tbody>
        </table>
    </div>
}
@if(technologyFormVisible) {
    <app-tecnologias-form [action]="action" [selectedTechnology]="selectedTechnology" (enabledTable)="enabled()" (getAll)="getAll()" (hiddenForm)="technologyFormVisible = $event"></app-tecnologias-form>
}
