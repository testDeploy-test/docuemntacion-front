import { Component, OnDestroy, OnInit, Input, Output, EventEmitter } from '@angular/core';
import { Chart, ChartType } from "chart.js/auto";
import { FormatoAuditoriaModel } from '../../../../models/formato-auditoria.model';
import { FormatoAuditoriaService } from '../../../../services/formato-auditoria.service';
import { AuditoriaDelProcesoModel } from '../../../../models/auditoria-del-proceso.model';
import { AuditoriaDelProcesoService } from '../../../../services/auditoria-del-proceso.service'; 
import { Subject } from 'rxjs';
import { ColorsService } from '../../../../services/colors.service';
import { takeUntil } from 'rxjs/operators';
import { FilterService } from '../../../../services/filter.service';

/**
 * Componente de la gráfica `Relación Item-Defectos`
 * @implements {OnDestroy}
 * @implements {OnInit}
 */
@Component({
  selector: 'app-items-defects-graphics',
  standalone: true,
  imports: [],
  templateUrl: './items-defects-graphics.component.html',
  styleUrl: './items-defects-graphics.component.css'
})
export class ItemsDefectsGraphicsComponent implements OnInit, OnDestroy {
  /**
   * Subject utilizado para desubscribirse de los observables al destruir el componente
   * @private
   * @type {Subject<void>}
   */
  private destroy$ = new Subject<void>();

  /**
   * Al inicializar el componente obten los registros de `formato_auditoria` y
   * `auditoria_del_proceso`
   * @method
   */
  ngOnInit(): void {
    this.getAudits();
    this.getProcesses();
  }

  /**
   * Al destruir el componente completa el subject para desubcribirse del observable
   * @method
   */
  ngOnDestroy(): void {
    this.destroy$.next();
    this.destroy$.complete();
  }

  /**
   * @constructor
   * @param {AuditoriaDelProcesoService} processService Servicio de la tabla `auditoria_del_proceso`
   * @param {FormatoAuditoriaService} auditService Servicio de la tabla `formato_auditoria`
   * @param {ColorsService} colorsService Servicio para proporcionar colores aleatorios
   * @param {FilterService} filterService Servicio del filtro
   */
  constructor(private processService: AuditoriaDelProcesoService, private auditService: FormatoAuditoriaService, private colorsService: ColorsService, private filterService: FilterService) {
    this.filterService.$emitter.subscribe(() => {
      this.filter();
    })
  }

  @Input() fecha1: string = ""; // Parámetro enviado desde el padre. Utilizado para el filtrado (inicio)
  @Input() fecha2: string = ""; // Parámetro enviado desde el padre. Utilizado para el filtrado (final)

  @Output() chartVisible = new EventEmitter<boolean> // Evento para cambiar la visibilidad del componente

  processList?: AuditoriaDelProcesoModel[]; // Declara la lista de registros de `auditoria_del_proceso` (modificable)
  processListOriginal?: AuditoriaDelProcesoModel[]; // Declara la lista de registros de `auditoria_del_proceso`
  auditsList?: FormatoAuditoriaModel[]; // Declara l alista de `formato_auditoria`
  list: any[] = []; // Declara la lista donde se almacenara la información en crudo necesaria para el gráfico

  chart: any; // Declara el gráfico
  itemsList: string[] = []; // Declara la lista de los items
  repeatList: number[] = []; // declara la lista de la cantidad total de defectos encontrados en los items

  /**
   * Método para obtener todos los registros de `formato_auditoria`.
   * Se desuscribe la observable al destruir el componente
   * @method
   */
  getAudits(): void {
    this.auditService.getAllAudits()
      .pipe(takeUntil(this.destroy$))
      .subscribe({
        next: (data) => {
          this.auditsList = data;
          this.createItemDefectsData();
        },
        error: (e) => console.error(e)
      });
  }

  /**
   * Método para obtener todos los registros de `auditoria_del_proceso`.
   * Se desuscribe la observable al destruir el componente
   * @method
   */  
  getProcesses(): void {
    this.processService.getAllProcesses()
      .pipe(takeUntil(this.destroy$))
      .subscribe({
        next: (data) => {
          this.processList = data;
          this.processListOriginal = data;
          this.createItemDefectsData();
        },
        error: (e) => console.error(e)
      });
  }

  /**
   * Método para obtener la información necesaria para la gráfica.
   * Primero llena `list` con la información relevante para este caso de `formato_auditoria` 
   * y `auditoria_del_proceso`. Después agrega los items a `itemsList` y la cantidad total de defectos en cada item 
   * agregala a `repeatList`
   * @method
   * @returns {void} Si `auditsList` y `processList` existen regresa una vez completada toda la operación para no ocultar el 
   * gráfico
   */
  createItemDefectsData(): void {
    const itemsRepetitions = new Map<string, number>();

    if(this.auditsList && this.processList) {

      for(let i = 0; i < this.auditsList.length; i++) {
        for(let j = 0; j < this.processList.length; j++) {
          if(this.auditsList[i].IDauditoria === this.processList[j].formato_auditorium.IDauditoria) {
            this.list.push({ item: this.auditsList[i].codigoItem, defects: this.processList[j].defectos_encontrados })
          }
        }
      }

      for(const info of this.list) {
        const item = info.item;
        const qunatity = info.defects;

        const insert1 = item ? item : "";
        const insert2 = qunatity ? qunatity : 0;

        if(itemsRepetitions.has(insert1)) {
          itemsRepetitions.set(insert1, itemsRepetitions.get(insert1)! + insert2)
        } else {
          itemsRepetitions.set(insert1, insert2)
        }
      }

      const array = Array.from(itemsRepetitions).sort((a, b) => b[1] - a[1])

      array.forEach((value) => {
        this.itemsList.push(value[0])
        this.repeatList.push(value[1])
      })
      
      this.createItemDefectsChart();
      return;
    }
    this.chartVisible.emit(false);
  }

  /**
   * Método para la creación del gráfico correspondiente.
   * Se especifica el tipo, y la información correspondiente extraida con el método `createItemDefectsData` además
   * de que se establecen unas configuraciónes del gráfico
   * @method
   */
  createItemDefectsChart() {
    this.chart = new Chart("ItemDefectsChart", {
      type: "bar" as ChartType,
      data: {
        labels: this.itemsList,
        datasets: [{
          data: this.repeatList,
          backgroundColor: this.colorsService.getColors(this.itemsList.length)
        }]
      },
      options: {
        responsive: true,
        plugins: {
          legend: { display: false },
          title: { display: true, text: "Relación Item-defectos" }
        },
        scales: {
          x: {
            title: { display: true, text: "Código item" }
          },
          y: {
            ticks: { stepSize: 1 },
            title: { display: true, text: "Total de defectos encontrados" }
          }
        },
        maintainAspectRatio: false
      }
    })
  }

  /**
   * Método para filtrar los registros por las fechas proporcionadas por el componente padre
   * `fecha1` y `fecha2`
   * @method
   * @returns {void} Regresa si no hay registros coincidentes con los parámetros establecidos u retorna
   * la ejecución del método `getProcesses` si los parámetros están vacíos
   */  
  filter() {
    // Destruye el gráfico actual
    this.chart.destroy();
    // Vacía las listas de los datos
    this.repeatList = [];
    this.itemsList = [];
    this.processList = [];
    this.list = [];
    // Haz visible el gráfico
    this.chartVisible.emit(true)

    // Si los parámetros de filtrado están vacíos retorna la ejecución de `getProcesses` 
    if(this.fecha1 === "" && this.fecha2 === "") {
      return this.getProcesses()
    }
    // Filtra la información según los parámetros
    const data = this.filterService.filterData(this.processListOriginal, { fecha1: this.fecha1, fecha2: this.fecha2 })
    // Si no hay registros que cumplan con los parámetros, oculata el gráfico y regresa
    if(data.length === 0) {
      this.chartVisible.emit(false);
      return;
    }

    // Si existen registros que cumplan los parámetros asignalos a `processList` y vuleve a generar la información pertinente
    this.processList = data;
    this.createItemDefectsData();
  }

}
