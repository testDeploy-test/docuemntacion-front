import { Component, ElementRef, Input, Output, OnInit, Renderer2, HostListener, EventEmitter } from '@angular/core';
import { Router, RouterLink } from '@angular/router';
import { AuthService } from '../../services/auth.service';
import { GetTablesService } from '../../services/get-tables.service';
import { provideIcons, NgIconComponent } from '@ng-icons/core';
import {heroArrowLeftStartOnRectangle as exit, heroHome} from "@ng-icons/heroicons/outline"
import { AlertsService } from '../../services/alerts.service';

/**
 * Componente del menú
 * @implements {OnInit}
 */
@Component({
  selector: 'app-navbar',
  standalone: true,
  imports: [NgIconComponent, RouterLink],
  viewProviders: [provideIcons({exit, heroHome})],
  templateUrl: './navbar.component.html',
  styleUrl: './navbar.component.css'
})
export class NavbarComponent implements OnInit {

  @Input() option: number = 0; // Opción para seleccionar entre `tablas` y `gráficas`
  @Input() table: string = ""; // Nombre de la tabla actual
  @Input() sideBar: boolean = true; // Visibilidad del sidebar
  @Output() hiddenBackgroundOfTables = new EventEmitter<boolean> // Visibilidad del fondo de `tablas`

  /**
   * Oculta el color de fondo de la pestaña `tablas`
   */
  changeValueOutput() {
    this.hiddenBackgroundOfTables.emit(false)
  }

  /**
   * Al inicializar el componente obtine una lista de las tablas, si se esta visualizando una tabla y 
   * se recarga la página le asigna un color para resaltado y evalua el tamaño de la pantalla
   * @method
   */
  ngOnInit(): void {
    this.getTables();
    this.addClass(this.option);
    this.getScreenSize()
  }

  showMenu: boolean = false; // Visiblidad del menu (para no mostrarlo en cietos componentes)
  tablesList?: any[]; // listado de las tablas
  //tablas a excluir, estas no son necesarias mostrarlas
  unionTables: string[] = ["proceso_defecto", "pedido_item"]
  //lista de tablas necesarias
  tablesArray:any;

  mobile: boolean = false; // evalua si la resolución es de celular

  /**
   * @constructor
   * @param {AuthService} authService Servicio de autencticación
   * @param {GetTablesService} tablesService Servicio para obtener el nombre de las tablas.
   * @param {Router} router Router
   * @param {Renderer2} renderer Servicio para renderizado en el DOM
   * @param {ElementRef} el Servicio para seleccionar elementos
   * @param {AlertsService} alertsService Servicio de alertas
   */
  constructor(
    private authService: AuthService, 
    private tablesService: GetTablesService,
    private router: Router,
    private renderer: Renderer2,
    private el: ElementRef,
    private alertsService: AlertsService
  ) {}

  /**
   * Metódo para cerrar la sesión del usuario. Redirige a `login` una vez cerrada la sesión, si 
   * falla imprime el error en la consola y lanza una alerta de error. Antes de cerrar sesión pide
   * consfirmación de la acción
   * @method
   */
  onSubmitLogout() {
    // Alerta de confirmación
    this.alertsService.confirmAlert("You are about to log out!", "logout").then((confirmed) => {
      // Si se confirma la acción cierra la sessión del usuario
      if(confirmed) {
        this.authService.logout().subscribe({
          // Si la acción es éxitosa lanza una alerta de éxito y redirige al `login`
          next: () => {
            this.alertsService.succesAlert("Logout successful");
            setTimeout(() => {
              this.router.navigate(["/login"])
            },1000)
          },
          // Si falla imprime el error en la consola y lanza una alerta de error.
          error: (e) => {
            console.log(e);
            this.alertsService.errorAlert()
          }
        })
      }
    })
  }


  /**
   * Obtiene un listado de las tablas a mostrar según el rol del usuario. Además remarca la 
   * tabla seleccionada si se reinicia la página.
   * @method
   */
  getTables() {
    // Obtiene las tablas
    this.tablesService.getTables().subscribe({
      next: (data) => {
        // Asigna un valor a `tablesList`
        this.tablesList = data as any[];
        // Evalúa si el usuario no es un administrador, de ser asi no mostrar la tabla `usuarios`
        if(!localStorage.getItem("isAdmin")) {
          this.tablesList = this.tablesList?.filter(table => table.nombre !== "usuarios")
        }
        // Asigna un valor a la lista de tablas a mostrar, excluyendo las tablas unión
        this.tablesArray = this.tablesList?.filter(table => !this.unionTables.includes(table.nombre))
        // Remarca la tabla seleccionada despues de 100ms
        setTimeout(() => {
          this.selectTable(this.table)
        }, 100)
      }, 
      // Si ocurre un error muestralo en conosla.
      error: (e) => console.error(e)
    })
  }

  /**
   * Resalta la opción seleccionada del menú.
   * @method
   * @param {Number} option Número asignado a cada elemento del menú.
   */
  addClass(option: number) {
    // Selecciona ambas opciones dl menu
    const graphics = this.el.nativeElement.querySelector(".graphics")
    const tables = this.el.nativeElement.querySelector(".tables");

    // Si `option` es 1 (tablas), la clase "element-selected" solo la debe tener `tables`
    if(option === 1) {
      this.renderer.removeClass(graphics, "element-selected")
      this.renderer.addClass(tables, "element-selected");
      
      // Si la resolución es de celular oculata el sidebar
      if(this.mobile) {
        this.sideBar =! this.sideBar
      }
    }
    // Si `option` es 2 (gráficas), la clase "element-selected" solo la debe tener `graphics`
    if(option === 2) {
      this.renderer.removeClass(tables, "element-selected");
      this.renderer.addClass(graphics, "element-selected");
    }
  }

  /**
   * Si se regresa o avanza en las pestañas utilizando las flechas del navegador, resalta
   * la tabla correspondiente
   * @method
   * @param {any} event Evento recibido
   */
  @HostListener('window:popstate', ['$event'])
  onPopState(event: any): void {
    const currentPath = window.location.pathname;
    const partes = currentPath.split("/").filter(part => part !== "")
    this.selectTable(partes[1])
  }
  /**
   * Obtiene el tamaño de la pantalla y si es una resolución de celular establece `mobile` como true y
   * oculta el sidebar.
   * @method
   * @param {any} event Evento recibido
   */
  @HostListener('window:resize', ['$event'])
  getScreenSize(event?: any) {
      const screenWidth = window.innerWidth;
      if(screenWidth <= 490) {
        this.mobile = true;
        this.sideBar = false;
      } else {
        this.mobile = false;
        // !! ocurre un error al cambiar el tamaño de la pantalla !!
      }
    }

  /**
   * Añade una clase a la tabla seleccionada
   * @param {any} table Nombre de la tabla seleccionada
   * @returns {void} Tiene returns para terminar la función en ciertos casos, como, si no se 
   * proporciona una tabla o `mobile` es true.
   */
  selectTable(table: any) {
    // Regresa si no hay tabla
    if(!table) return;
    // Si es el tamaño de la pantalla es de celular, oculata el sidebar, el color de fondo y regresa.
    if(this.mobile) {
      this.sideBar = false;
      this.changeValueOutput()
      return;
    }
    // Elemento con la clase de la tabla proporcionada
    const selected = this.el.nativeElement.querySelector(`.${table}`);
    // Si existe el elemento de la tabla...
    if (selected) {
      // Oculta el color de fondo.
      this.changeValueOutput()
      // itera en cada tabla de `tablesList`
        for (const tab of this.tablesList || []) {
          // Remueve la clase `table-selected` de las tablas que no sean la seleccionada.
            if (tab.nombre !== table) {
                const other = this.el.nativeElement.querySelector(`.${tab.nombre}`);
                if (other) {
                    this.renderer.removeClass(other, "table-selected");
                }
            }
        }
        // Añade la clase `table-selected` a la tabla seleccionada.
        this.renderer.addClass(selected, "table-selected");
    } else {
      // Imprimer un error en consola si no existe un elemento con la clase proporcionado
        console.error(`Element with class "${table}" not found.`);
    }
}
}

    <div class="tablesList">

        <ul class="tables-container">
            @for(table of tablesArray; track table.id) {
                    <li routerLink="/tablas/{{table.nombre}}" (click)="selectTable(table.nombre)"

            }

}
