import { Component, OnInit, Input, Output, EventEmitter } from '@angular/core';
import { PedidosService } from '../../../services/pedidos.service';
import { PedidoItemService } from '../../../services/pedido-item.service';
import { PedidoItemModel } from '../../../models/pedido-item.model';
import { ItemsService } from '../../../services/items.service';
import { ItemsModel } from '../../../models/items.model';
import { TecnologiasService } from '../../../services/tecnologias.service';
import { TecnologiasModel } from '../../../models/tecnologias.model';
import { ClientesService } from '../../../services/clientes.service';
import { ClientesModel } from '../../../models/clientes.model';
import { AlertsService } from '../../../services/alerts.service';
import { FormArray, ReactiveFormsModule } from '@angular/forms';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import { provideIcons, NgIconComponent } from '@ng-icons/core';
import { heroXMark } from '@ng-icons/heroicons/outline';
import { Subscription } from 'rxjs';

/**
 * Componente del formulario de `pedidos`
 * @implements {OnInit}
 */
@Component({
  selector: 'app-pedidos-form',
  standalone: true,
  imports: [ReactiveFormsModule, NgIconComponent],
  viewProviders: [provideIcons({heroXMark})],
  templateUrl: './pedidos-form.component.html',
  styleUrl: './pedidos-form.component.css'
})
export class PedidosFormComponent implements OnInit {

  /**
   * Al inicializar el componente crea el formulario dinámico, obten la información de
   * `tecnologias` y `pedido_item` y suscribe el resultado de estas a la vusalización del 
   * formulario según la acción especificada en el padre, además obtén los registros de 
   * `items` y `clientes`
   * @method
   */
  ngOnInit(): void {
    this.createDinamicForm();
    this.getTechnologies().add(() => {
      this.actionInit();
    })
    this.getOI().add(() => {
      this.actionInit();
    })
    this.getItems()
    this.getCustomers()
  }

  @Input() selectedOrder: any; // Recibe la información del registro seleccionado a actualizar
  @Input() action: string = ""; // Recibe la acción del formulario
  
  @Output() getAll = new EventEmitter<void> // Evento para activar el método `getAll` del padre
  @Output() getPI = new EventEmitter<void> // Evento para activar el método `getPI` del padre
  @Output() hiddenForm = new EventEmitter<boolean> // Evento para cambiar la visibilidad del formulario en el padre
  @Output() enabledTable = new EventEmitter<void> // Evento para activar el método `enabled` del padre

  oiList?: PedidoItemModel[]; // Declara la lista de registros de `pedido_item`
  itemsList?: ItemsModel[]; // Declara la lista de registros de `items`
  technologiesList?: TecnologiasModel[]; // Declara la lista de registros de `tecnologias`
  customersList?: ClientesModel[]; // Declara la lista de registros de `clientes`

  itemsRemoved: any[] = []; // Declara la lista de items elimindos del formulario

  ordersForm: FormGroup; // Declara el formulario principal
  oiForm: any; // Declara el formulario secundario
  oiFormArray: FormGroup; // Declara el formulario array con elementos de `oiForm`

  selectedFile: File | any; // Declara la imagen

  selectedOI: any; // Declara la información de `pedido_item` según el pedido seleccionado
  duplicated: boolean = false; // Para verificar si hay items repetidos en el pedido

  /**
   * @constructor
   * @param {FormBuilder} fb Servicio para crear el formulario
   * @param {PedidosService} ordersService Servicio de la tabla `pedidos`
   * @param {PedidoItemService} oiService Servicio de la tabla `pedido_item`
   * @param {TecnologiasService} technologiesService Servicio de la tabla `tecnologias`
   * @param {ItemsService} itemsService Servicio de la tabla `items`
   * @param {ClientesService} customersService Servicio de la tabla `clientes`
   * @param {AlertsService} alertsService Servicio de alertas
   */
  constructor(
    private fb: FormBuilder,
    private ordersService: PedidosService,
    private oiService: PedidoItemService,
    private technologiesService: TecnologiasService,
    private itemsService: ItemsService,
    private customersService: ClientesService,
    private alertsService: AlertsService
  ) {
    // Define los campos del formulario principal
    this.ordersForm = this.fb.group({
      codigoPedido: ["", Validators.required],
      descripcion: ["", Validators.required],
      IDtecnologia: ["", Validators.required],
      codigoCliente: ["", Validators.required],
      file: [null, Validators.required]
    })

    // Define los campos del formulario array
    this.oiFormArray = this.fb.group({
      data: this.fb.array([])
    })
  }

  /**
   * Método para identificar la acción del formulario y en base a esto
   * cambiar su presentación
   * @method
   */
  actionInit(): void {
    if(this.action === "update") {
      this.showFormUpdate()
    } else {
      this.showFormCreate()
    }
  }

  /**
   * Método para inicializar el formulario array
   * @method
   */
  createDinamicForm(): void {
    this.oiFormArray;
  }

  /**
   * Facilita la obtención de la información del formulario array
   */
  get formArray(): FormArray {
    return this.oiFormArray.get("data") as FormArray
  }

  /**
   * Método para agregar un elemento `oiForm` al formulario array
   * @method
   */
  addOI(): void {
    this.oiForm = this.fb.group({
      codigoPedido: [this.ordersForm.get("codigoPedido")?.value, Validators.required],
      codigoItem: ["", Validators.required],
      cantidad: ["", Validators.required]
    })
    this.formArray.push(this.oiForm)
  }

  /**
   * Método para eliminar un elemento del formulario array
   * @method
   * @param {Number} index index del elemento a eliminar
   * @returns {void} Retorna si el index es inválido
   */
  deleteOI(index: number): void {
    if(index < 0 || index > this.formArray.length) {
      this.alertsService.errorAlert();
      return;
    }
    //no funiona de forma normal, encontré lo siguente para solucionarlo
    const value = this.formArray.value;

    this.formArray.setValue(
      value.slice(0, index).concat(
        value.slice(index + 1),
      ).concat(value[index]),
    );

    if(this.action === "update") {
      this.itemsRemoved.push(this.formArray.at(value.length - 1).value)
    }
    this.hasDuplicated(null);
    this.formArray.removeAt(value.length - 1);

  }

  /**
   * Método que verifica si hay elementos en el formulario array con el mismo
   * `codigoItem` y `codigoPedido`, este último si se esta actualizando un registro
   * @method
   */
  public hasDuplicated(event: any): void {
    let count: number = 0;
    this.duplicated = false; 
    if(event != null) {
      for (let i = 0; i < this.formArray.length; i++) {
        if (event.target.value === this.formArray.at(i).value.codigoItem) {
            count++;
            if (count >= 2) {
                this.duplicated = true;
            }
        }
    }
    }
    count = 0;
  }

  /**
   * Método para obtener todos los registros de `tecnologias` y si es éxitosa asignar la respuesta
   * a `technologiesList` y si falla mostrar el error en la consola
   * @method
   * @returns {Subscription} Suscripción de la infromación recabada en este método para su 
   * futura utilización en alguno de los métodos de visualización de formularios
   */
  public getTechnologies(): Subscription {
    return this.technologiesService.getAllTechnologies().subscribe({
      next: (data) => {
        this.technologiesList = data
      },
      error: (e) => console.log(e)
    })
  }

  /**
   * Método para obtener todos los registros de `items` y si es éxitosa asignar la respuesta
   * a `itemsList` y si falla mostrar el error en la consola
   * @method
   */
  public getItems(): void {
    this.itemsService.getAllitems().subscribe({
      next: (data) => {
        this.itemsList = data 
      },
      error: (e) => console.error(e)
    })
  }

  /**
   * Método para obtener todos los registros de `clientes` y si es éxitosa asignar la respuesta
   * a `customersList` y si falla mostrar el error en la consola
   * @method
   */  
  public getCustomers(): void {
    this.customersService.getAllCustomers().subscribe({
      next: (data) => {
        this.customersList = data;
      },
      error: (e) => console.log(e)
    })
  }

  /**
   * Método para obtener ttodos los registros de `pedido_item` y si es éxitosa asignar la respuesta
   * a `oiList` y si falla mostrar el error en la consola
   * @method
   * @returns {Subscription} Suscripción de la infromación recabada en este método para su 
   * futura utilización en alguno de los métodos de visualización de formularios
   */  
  public getOI(): Subscription {
    return this.oiService.getAllOI().subscribe({
      next: (data) => {
        this.oiList = data
      },
      error: (e) => console.error(e)
    })
  }

  /**
   * Método para asignar el valor del campo file a `ordersForm`
   * @method
   * @param {any} event Input imagen
   */
  onFileSelected(event: any): void {
    this.selectedFile = event.target.files[0];
    this.ordersForm.patchValue({
      file: this.selectedFile
    })
  }

  /**
   * Método que vacía los campos del formulariocuando se visualiza para crear un nuevo registro
   * @method
   */  
  public showFormCreate(){
    this.ordersForm.setValue({
      codigoPedido: null,
      descripcion: null,
      IDtecnologia: null,
      codigoCliente: null,
      file: null
    })
  }

  /**
   * Método para establecer el valor del registro a actualizar en los campos correspondientes de los formularios
   * @method
   */  
  public showFormUpdate() {
    const id = this.technologiesList?.find(tech => tech.nombre === this.selectedOrder.tecnologia.nombre)
    this.ordersForm.setValue({
      codigoPedido: this.selectedOrder.codigoPedido,
      descripcion: this.selectedOrder.descripcion,
      IDtecnologia: id?.IDtecnologia,
      codigoCliente: this.selectedOrder.codigoCliente,
      file: this.selectedOrder.imagen
    })
    // En base al pedido seleccionado inserta los registros de `pedido_item` que tengan el mismo `codigoPedido` al formulario array 
    this.selectedOI = this.oiList?.filter(oi => oi.codigoPedido === this.selectedOrder.codigoPedido)
    if(this.selectedOI) {
      for(let i = 0; i < this.selectedOI?.length; i++){
        this.formArray.push(this.fb.group({
          IDpedido_item: this.selectedOI[i].IDpedido_item,
          codigoPedido: this.selectedOI[i].codigoPedido,
          codigoItem: this.selectedOI[i].item.codigoItem,
          cantidad: this.selectedOI[i].cantidad
        }))
      }
    }
  }

  /**
   * Método para crear un nuevo registro de `pedidos` en base a la información recabada en el formulario convertido a `FormData`.
   * Si es éxitosa llama al método `createOI`, muestra una alerta de éxito, llama a los métodos `hidden` y `getInfo`
   * y si falla muestra el error en la consola y muestra una alerta de error 
   * @method
   * @returns {void} Si el formulario es inválido envia una alerta para que complete
   * el formulario y retorna
   */  
  public create(): void {
    if(this.ordersForm.invalid) {
      this.alertsService.completeAlert();
      return;
    }

    const formData = new FormData();
    formData.append("codigoPedido", this.ordersForm.get("codigoPedido")?.value);
    formData.append("descripcion", this.ordersForm.get("descripcion")?.value);
    formData.append("IDtecnologia", this.ordersForm.get("IDtecnologia")?.value);
    formData.append("codigoCliente", this.ordersForm.get("codigoCliente")?.value);
    formData.append("file", this.selectedFile, this.selectedFile.name)

    this.ordersService.createOrder(formData).subscribe({
      next: () => {
        this.createOI();
        this.alertsService.succesAlert("New order created");
        this.hidden();
        this.getInfo();
      },
      error: (e) => {
        console.error(e);
        this.alertsService.errorAlert()
      }
    })

  }

  /**
   * Método para crear nuevos registros de `pedido_item` en base a la información recabada 
   * en el formulario array. Si es éxitoso muestra en consola un mensaje de éxito y si falla
   * muestra un mensaje de error en la consola y lanza una alerta de error
   * @method
   */
  public createOI(): void {
    this.oiService.createOI(this.formArray.value).subscribe({
          next: () => {
            console.log("new order - items created")
          }, 
          error: (e) => {
            console.error(e);
            this.alertsService.errorAlert()
          }
        })
  }

  /**
   * Método para actualizar un registro de `pedidos` en base a `codigoPedido` del registro seleccionado
   * y en la nueva información proporcionada en el formulario.
   * Pide confirmación de la acción. En caso de confirmar y que la acción sea éxitosa llama al método `updateOI`,
   *  muestra una alerta de éxito, llamada a los métodos `hidden` y `getInfo` y si falla muestra el error en la 
   * consola y lanza una alerta de error
   * @method
   */
  public update(): void {
    this.alertsService.confirmAlert("You are about to update a record.", "update").then((confirmed) => {
      if(confirmed) {
        const codigoPedido = this.selectedOrder.codigoPedido;

        const formData = new FormData;
        formData.append("codigoPedido", this.ordersForm.get("codigoPedido")?.value);
        formData.append("descripcion", this.ordersForm.get("descripcion")?.value);
        formData.append("IDtecnologia", this.ordersForm.get("IDtecnologia")?.value);
        formData.append("codigoCliente", this.ordersForm.get("codigoCliente")?.value);
        if(this.selectedOrder.imagen !== this.ordersForm.get("file")?.value) {
          formData.append("file", this.selectedFile, this.selectedFile.name)
        }
        this.ordersService.updateOrder(codigoPedido, formData).subscribe({
          next: () => {
            this.updateOI();
            this.alertsService.succesAlert(`Record ${codigoPedido} updated.`);
            this.hidden();
            this.getInfo();
          },
          error: (e) => {
            console.error(e);
            this.alertsService.errorAlert()
          }
        })
      }
    })

  }

  /**
   * Método para actualizar registros de `pedido_item` en base a `IDpedido_item`.
   * Si en la actualización del pedido se agregaron más items, crea los faltantes.
   * Si en la actualización del pedido Se cambiaron los registros existentes actualizalos uno por uno con la nueva información.
   * Si en la actualización del pedido se eliminaron registros eliminalos uno por uno
   * @method
   * @returns {void} Regresa si los registros de `pedido_item` y el formulario array son iguales
   */
  public updateOI(): void {
    // Si no hay cambios en la información regresa
    if(JSON.stringify(this.selectedOI) === JSON.stringify(this.formArray.value)) {
      return;
    }
    // Si se agregaron elementos, crealos
    if(this.selectedOI.length < this.formArray.length) {
      const data = this.formArray.value.filter((oi: any) => !oi.IDpedido_item)
      this.oiService.createOI(data).subscribe({
        next: () => {
          console.log("New records created.")
        },
        error: (e) => {
          console.error(e);
          this.alertsService.errorAlert()
        }
      })
    }
    // Si se actualizaron, modificalos uno por uno
    for(let i = 0; i < this.selectedOI.length; i++) {
      if(JSON.stringify(this.selectedOI[i]) !== JSON.stringify(this.formArray.at(i)?.value)) {
        this.oiService.updateOI(this.selectedOI[i]?.IDpedido_item, this.formArray.at(i)?.value).subscribe({
          next: () => {
            console.log(`Record ${i + 1} updated.`)
          },
          error: (e) => {
            console.error(e);
            this.alertsService.errorAlert()
          }
        })
      }
    }

    // Si se eliminaron registros, eiminalos uno por uno
    if(this.itemsRemoved.length !== 0){
      for(let i = 0; i < this.itemsRemoved.length; i++){
        const IDpedido_item = this.itemsRemoved[i].IDpedido_item;
        if(IDpedido_item !== "") {
          this.oiService.deleteOI(IDpedido_item).subscribe({
            next: () => {
              console.log(`Order - items ${IDpedido_item} removed.`)
            },
            error: (e) => {
              console.error(e);
              this.alertsService.errorAlert()
            }
          })
        }
      }
    }
  }

  /**
   * Método para cerrar el formulario detectando cambios en la información inicial tanto para la creación
   * de un nuevo registro o la actualización de un registro. En caso de detectar cambios pide una confirmación 
   * antes de cerrar el formulario en caso de confirmar llama al método `hidden`
   * @method
   * @returns Retorna si se cumplen ciertas condiciones para no ejecutar el resto de la lógica
   */  
  public close(): void {
    // Si la acción es para actualizar un registro...
    if(this.action === "update") {
      // Obtén el `IDtecnologia` de la tecnologia del pedido
      const id = this.technologiesList?.find(tech => tech.nombre === this.selectedOrder.tecnologia.nombre)
      
      // Estructura inicial de la información del registro a actualizar
      const orderPrev = this.selectedOrder? {
        codigoPedido: this.selectedOrder.codigoPedido,
        descripcion: this.selectedOrder.descripcion,
        IDtecnologia: id?.IDtecnologia,
        codigoCliente: this.selectedOrder.codigoCliente,
        file: this.selectedOrder.imagen
      } : null

      // Estructura inicial de la información del formulario array
      this.selectedOI = this.oiList?.filter(oi => oi.codigoPedido === this.selectedOrder.codigoPedido)
      const oiPrev = this.selectedOI?.map((oi: any) => ({
        IDpedido_item: oi.IDpedido_item,
        codigoPedido: oi.codigoPedido,
        codigoItem: oi.item.codigoItem,
        cantidad: oi.cantidad
      }));
  
      // Si los datos previos y actuales son iguales llama al método `hidden` y regresa
      if((JSON.stringify(this.ordersForm.value) === JSON.stringify(orderPrev)) && (JSON.stringify(this.formArray.value) === JSON.stringify(oiPrev))) {
        this.hidden();
        return;
      }

      // Si los datos previos y actuales son diferentes pide confirmación para cerrar el formulario
      this.alertsService.confirmAlert("The changes will be lost.", "close").then((confirmed) => {
        if(confirmed) {
          this.hidden();
          return;
        }
      })
      return;
    }

    // Si la acción es para crear un nuevo registro...
    // Estructura inicial de la información del formulario
    const newOrder = {
      codigoPedido: null,
      descripcion: null,
      IDtecnologia: null,
      codigoCliente: null,
      file: null
    }

    // Si los datos previos y actuales son iguales llama al método `hidden` y regresa
    if(JSON.stringify(this.ordersForm.value) === JSON.stringify(newOrder) && !this.formArray.at(0)) {
      this.hidden();
      return;
    }
    // Si los datos previos y actuales son diferentes pide confirmación para cerrar el formulario
    this.alertsService.confirmAlert("The data will be lost.", "close").then((confirmed) => {
      if(confirmed) {
        this.hidden();
        return;
      }
    })

  }

  /**
   * Método para obtener la información necesaria para el componente pedidos
   * mediante la emisión de lo eventos `getAll` y `getPI`
   * @method
   */
  getInfo() {
    this.getAll.emit();
    this.getPI.emit();
  }

  /**
   * Método para ocultar el formulario y habilitar los botones de las acciónes en el componente padre
   * emitiendo los eventos `hiddenForm` y `enabledTable`
   * @method
   */    
  hidden() {
    this.hiddenForm.emit(false);
    this.enabledTable.emit();
  }
 
}
