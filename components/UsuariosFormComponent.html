import { Component, Input, Output, EventEmitter, OnInit } from '@angular/core';
import { ReactiveFormsModule } from '@angular/forms';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import { UsuariosService } from '../../../services/usuarios.service';
import { provideIcons, NgIconComponent } from '@ng-icons/core';
import { heroXMark } from '@ng-icons/heroicons/outline';
import { AlertsService } from '../../../services/alerts.service';

/**
 * Componente del formulario de `usuarios`
 * @implements {OnInit}
 */
@Component({
  selector: 'app-usuarios-form',
  standalone: true,
  imports: [ReactiveFormsModule, NgIconComponent],
  viewProviders: [provideIcons({heroXMark})],
  templateUrl: './usuarios-form.component.html',
  styleUrl: './usuarios-form.component.css'
})
export class UsuariosFormComponent implements OnInit{

  /**
   * Al inicializar el componente muestra el formulario según la acciónespecificada en el componente padre
   * @method
   */
  ngOnInit(): void {
    if(this.action === "update") {
      this.showFormUpdate()
    } else {
      this.showFormCreate()
    }
  }

  @Input() userSelected: any; // Recibe la información del registro a actualizar
  @Input() action: string = ""; // Recibe la acción del formulario

  @Output() getAll = new EventEmitter<void> // Evento para activar el método `getAll` del padre
  @Output() hiddenForm = new EventEmitter<boolean> // evento para cambiar la vsibilidad del formalario en el padre
  @Output() enabledTable = new EventEmitter<void> // Evento para activar el método `enabled` del padre

  public usersForm: FormGroup; // Declara el formulario
  
  /**
   * @constructor
   * @param {FormBuilder} fb Servicio para crear el formulario
   * @param {AlertsService} alertsService Servicio de alertas
   * @param {UsuariosService} usersService Servicio de la tabla `usuarios`
   */
  constructor(
    private fb: FormBuilder,
    private alertsService: AlertsService,
    private usersService: UsuariosService
  ) {
    // define los campos del formulario
    this.usersForm = this.fb.group({
      nombre: ["", Validators.required],
      usuario: ["", Validators.required],
      contraseña: ["", Validators.required],
      activo: ["", Validators.required],
      superAdmin: ["", Validators.required],
      hora_inicio: ["", Validators.required],
      hora_fin: ["", Validators.required]
    })
  }

  /**
   * Método para formatear los campos `hora_inicio` y `hora_fin` a un formato aceptable
   * @method
   * @param {Number} option Acción a realizar 1 si es creación y 2 si es actualización
   * @returns {void} Retornna al cumplir las condiciones para no ejecutar el resto del código
   */
  public setHours(option: number): void {
    const hi = this.usersForm.get("hora_inicio")?.value;
    const hf = this.usersForm.get("hora_fin")?.value;
    if(option === 1) {  
      this.usersForm.patchValue({
        hora_inicio: hi + ":00",
        hora_fin: hf + ":00"
      })
      return;
    }
    if(option === 2){
      if(this.usersForm.get("hora_inicio")?.value !== this.userSelected.hora_inicio) {
        this.usersForm.patchValue({
          hora_inicio: hi + ":00"
        })
      }
      if(this.usersForm.get("hora_fin")?.value !== this.userSelected.hora_fin) {
        this.usersForm.patchValue({
          hora_fin: hf + ":00"
        })
      }
    }
  }

  /**
   * Método para crear un nuevo registro de `usuarios` en base a la información recabada en el formulario.
   * Si es éxitosa muestra una alerta de éxito, activa el evento `getAll` y llama al método `hidden`
   * y si falla muestra el error en la consola y muestra una alerta de error 
   * @method
   * @returns {void} Si el formulario es inválido envia una alerta para que complete
   * el formulario y retorna
   */  
  public create(): void {
    if(this.usersForm.invalid) {
      this.alertsService.completeAlert();
      return;
    }
    this.setHours(1)
    this.usersService.createUser(this.usersForm.value).subscribe({
      next: () => {
        this.alertsService.succesAlert("New record of Users created.");
        this.getAll.emit();
        this.hidden();
      },
      error: (e) => {
        console.error(e);
        this.alertsService.errorAlert()
      }
    })
  }

  /**
   * Método para actualizar un registro de `usuarios` en base a `IDauditoria` del registro seleccionado
   * y en la nueva información proporcionada en el formulario.
   * Pide confirmación de la acción. En caso de confirmar y que la acción sea éxitosa muestra una alerta de éxito,
   * emite el evento `getAll` y llamada al método `hidden` y si falla muestra el error en la consola y lanza
   * una alerta de error
   * @method
   */  
  public update(): void {
    this.alertsService.confirmAlert("You are about to update a record.", "update").then((confirmed) => {
      if(confirmed) {
        const IDusuario = this.userSelected.IDusuario;
        this.setHours(2);
        if(this.usersForm.get("contraseña")?.value === null) {
          this.usersForm.removeControl("contraseña")
        }
        this.usersService.updateUser(IDusuario, this.usersForm.value).subscribe({
          next: () => {
            this.alertsService.succesAlert(`Record ${IDusuario} updated.`);
            this.getAll.emit();
            this.hidden();
          },
          error: (e) => {
            console.error(e);
            this.alertsService.errorAlert()
          }
        })
      }
    })
  }

  /**
   * Método que vacía los campos del formulariocuando se visualiza para crear un nuevo registro
   * @method
   */  
  public showFormCreate(): void {
    this.usersForm.setValue({
      nombre: null,
      usuario: null,
      contraseña: null,
      activo: null,
      superAdmin: null,
      hora_inicio: null,
      hora_fin: null
    })
  }

  /**
   * Método para establecer el valor del registro a actualizar en los campos correspondientes del formulario
   * @method
   */  
  public showFormUpdate(): void {
    this.usersForm.setValue({
      nombre: this.userSelected.nombre,
      usuario: this.userSelected.usuario,
      contraseña: null,
      activo: this.userSelected.activo,
      superAdmin: this.userSelected.superAdmin,
      hora_inicio: this.userSelected.hora_inicio,
      hora_fin: this.userSelected.hora_fin
    })
  }

  /**
   * Método para cerrar el formulario detectando cambios en la información inicial tanto para la creación
   * de un nuevo registro o la actualización de un registro. En caso de detectar cambios pide una confirmación 
   * antes de cerrar el formulario en caso de confirmar llama al método `hidden`
   * @method
   * @returns Retorna si se cumplen ciertas condiciones para no ejecutar el resto de la lógica
   */  
  public close(): void {
    // Estructura inicial de la información de un nuevo registro
    const dataNew = {
      nombre: null,
      usuario: null,
      contraseña: null,
      activo: null,
      superAdmin: null,
      hora_inicio: null,
      hora_fin: null
    }

    // Estructura inicial de la información de un registro a actualizar
    const dataPrev = this.userSelected? {
      nombre: this.userSelected.nombre,
      usuario: this.userSelected.usuario,
      contraseña: null,
      activo: this.userSelected.activo,
      superAdmin: this.userSelected.superAdmin,
      hora_inicio: this.userSelected.hora_inicio,
      hora_fin: this.userSelected.hora_fin
    } : null

    // Si la acción es para ctualizar un formulario...
    if(this.action === "update") {
      // Si la información previa y actual es igual llama al método `hidden` y regresa
      if(JSON.stringify(this.usersForm.value) === JSON.stringify(dataPrev)) {
        this.hidden()
        return;
      }
      // Si la información previa y actual son diferentes pide confirmación para cerrar el formulario
      this.alertsService.confirmAlert("The changes will be lost.", "close").then((confirmed) => {
        if(confirmed) {
          this.hidden()
          return;
        }
      })
      return;
    }

    // Si la acción es para crear un nuevo registro...
    // Si la información previa y actual es igual llama al método `hidden` y regresa
    if(JSON.stringify(this.usersForm.value) === JSON.stringify(dataNew)) {
      this.hidden()
      return;
    }
    // Si la información previa y actual es diferente pide confirmación para cerrar el formulario
    this.alertsService.confirmAlert("The data will be lost.", "close").then((confirmed) => {
      if(confirmed) {
        this.hidden()
        return;
      }
    })
  }

  /**
   * Método para ocultar el formulario y habilitar los botones de las acciónes en el componente padre
   * emitiendo los eventos `hiddenForm` y `enabledTable`
   * @method
   */    
  public hidden() {
    this.hiddenForm.emit(false);
    this.enabledTable.emit();
  }

}
