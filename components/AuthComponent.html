import { Component, OnInit } from '@angular/core';
import { ReactiveFormsModule, FormBuilder, FormGroup, Validators } from '@angular/forms';
import { Title } from '@angular/platform-browser';
import { AuthService } from '../../services/auth.service';
import { provideIcons, NgIconComponent } from '@ng-icons/core';
import { heroUserCircle} from "@ng-icons/heroicons/outline"
import { AlertsService } from '../../services/alerts.service';
import { Router } from '@angular/router';
import { UsuariosService } from '../../services/usuarios.service';

/**
 * Componente del formulario de login
 * @implements {OnInit}
 */
@Component({
  selector: 'app-auth',
  standalone: true,
  imports: [NgIconComponent, ReactiveFormsModule],
  viewProviders: [provideIcons({heroUserCircle})],
  templateUrl: './auth.component.html',
  styleUrl: './auth.component.css'
})
export class AuthComponent implements OnInit {
  title = "Login";

  public loginForm: FormGroup;

  /**
   * @constructor
   * @param {Title} titleService 
   * @param {AuthService} authService 
   * @param {FormBuilder} fb 
   * @param {Router} router 
   * @param {AlertsService} alertsService 
   * @param {UsuariosService} userService 
   */
  constructor(
    private titleService: Title,
    private authService: AuthService, 
    private fb: FormBuilder, 
    private router: Router,
    private alertsService: AlertsService,
    private userService: UsuariosService
  ) {
    // Campos del formulario de login.
    this.loginForm = this.fb.group({
      usuario: ["", Validators.required],
      contraseña: ["", Validators.required]
    })
  }

  /**
   * Al inicializar el componente cambia el titulo.
   * @method
   */
  ngOnInit(): void {
    this.titleService.setTitle("Login")
  }

  /**
   * Se ejecuta al intentar enviar el formulario, el cual solo se enviará si es válido.
   * @method onSubmitLogin 
   */
  onSubmitLogin() {
    // Si el formulario es inválido, lanza una alerta para que el usuario lo complete.
    if(this.loginForm.invalid) {
      this.alertsService.completeAlert()
    } else {
      // Si el formulario es válido intenta inicisar sesión con los datos proporcionados.
      this.authService.login(this.loginForm.value).subscribe({
        /*  Si el login es éxitoso lanza una alerta de éxito, verifica si es un administrador y
        redirige al usuario al inicio
        */
        next: (res: any) => {
          this.alertsService.succesAlert("Login successful")
          this.isAdmin(res["id"]);
          setTimeout(() => {
            this.router.navigate(["/"]);
          }, 1000)
        },
        // Si el login falla imprime en cosola el error y lanza una alerta de error 
        error: (e) => {
          console.error(e);
          this.alertsService.errorAlert();
        }
      })
    }
  }

  /**
   * Verifica si el usuario que inició sesión es un administrador.
   * @method isAdmin
   * @param {Number} id ID del usuario
   */
  isAdmin(id: number): void {
    // Obten al usuario en base a su id.
    this.userService.getOneUser(id).subscribe({
      /* Si el usuario existe verifica si el capo superAdmin es true o false
      en caso de ser true establece un item llamado `isAdmin`
      */
      next: (res) => {
        if(res.superAdmin) {
          localStorage.setItem("isAdmin", "yes")
        } else {
          const admin = localStorage.getItem("isAdmin")
          admin ? localStorage.removeItem("isAdmin") : ""
        }
      },
      // Si ocurre un eror remieve el item `isAdmin` del localStorage, por si las dudas
      error: () => {
        localStorage.removeItem("isAdmin")
      }
    })
  }
}
