import { Component, OnInit } from '@angular/core';
import { Title } from '@angular/platform-browser';
import { FormsModule } from '@angular/forms';
import { NavbarComponent } from '../navbar/navbar.component';
import { FooterComponent } from '../footer/footer.component';
import { FormatoAuditoriaModel } from '../../models/formato-auditoria.model';
import { FormatoAuditoriaService } from '../../services/formato-auditoria.service';
import { AuditoriaDelProcesoModel } from '../../models/auditoria-del-proceso.model';
import { AuditoriaDelProcesoService } from '../../services/auditoria-del-proceso.service';
import { ProcesoDefectoModel } from '../../models/proceso-defecto.model';
import { ProcesoDefectoService } from '../../services/proceso-defecto.service';
import { PedidosModel } from '../../models/pedidos.model';
import { PedidosService } from '../../services/pedidos.service';
import { PedidoItemModel } from '../../models/pedido-item.model';
import { PedidoItemService } from '../../services/pedido-item.service';
import { ExcelService } from '../../services/excel.service';
import { Subscription } from 'rxjs';
import { provideIcons, NgIconComponent } from '@ng-icons/core';
import { bootstrapX, bootstrapArrowBarUp } from "@ng-icons/bootstrap-icons";
import { FilterService } from '../../services/filter.service';
import { AlertsService } from '../../services/alerts.service';

/**
 * Componente del inicio.
 * @implements {OnInit}
 */
@Component({
  selector: 'app-home',
  standalone: true,
  imports: [NavbarComponent, FooterComponent, FormsModule, NgIconComponent],
  viewProviders: [provideIcons({bootstrapX, bootstrapArrowBarUp})],
  templateUrl: './home.component.html',
  styleUrl: './home.component.css'
})
export class HomeComponent implements OnInit {
    
  /**
   * @constructor
   * @param {Title} titleService Servicio para interactuar con el titulo
   * @param {FormatoAuditoriaService} auditsService Servicio de `fomato_auditoria`
   * @param {AuditoriaDelProcesoService} processService Servicio de `auditoria_del_proceso`
   * @param {ProcesoDefectoService} pdService servicio de `proceso_defecto`
   * @param {PedidosService} ordersService Servicio de `pedidos`
   * @param {PedidoItemService} oiService Servicio de `pedido_item`
   * @param {ExcelService} excelService Servicio de Excel
   * @param {FilterService} filterService Servicio del Filtro
   * @param {AlertsService} alertsService Servicio de aalertas
   */
  constructor(
    private titleService: Title,
    private auditsService: FormatoAuditoriaService,
    private processService: AuditoriaDelProcesoService,
    private pdService: ProcesoDefectoService,
    private ordersService: PedidosService,
    private oiService: PedidoItemService,
    private excelService: ExcelService,
    private filterService: FilterService,
    private alertsService: AlertsService
  ) {}

  /**
   * Al inicializar el componete obtner la información de las tablas a exportar a Excel y
   * que las métodos de esto subscriban a las métodos para formatear la información para que no haya 
   * problemas en la disponibilidad de la información.
   */
  ngOnInit(): void {
    // Establece un titulo al inicializar el componente.
    this.titleService.setTitle("Home");

    // Extrae los datos de `formato_auditoria`
    this.getAuditsToExcel().add(() => {
      this.dataOfAudits()
    })
    //Extrae los datos de `auditoria_del_proceso`
    this.getProcessToExcel().add(() => {
      this.dataOfProcess()
    })
    // Extrae los datos de `pedidos`
    this.getOrdersToExcel().add(() => {
      this.dataOfOrders()
    })
  }
  

  auditsList?: FormatoAuditoriaModel[]; // Inicializamos la lista de `formato_auditoria`
  processList?: AuditoriaDelProcesoModel[]; // Inicializamos la lista de `auditoria_del_proceso`
  ordersList?: PedidosModel[]; // Inicializamos la lista de `pedidos`
  pdList?: ProcesoDefectoModel[]; // // Inicializamos la lista de `proceso_defecto`
  oiList?: PedidoItemModel[]; // Inicializamos la lista de `pedido_item`

  fecha1: string = ""; // Primer propiedad del segundo parámetro de filterData - fecha inicial
  fecha2: string = ""; // Segunda propiedad del segundo parámetro de filterData - fecha de finalización

  buttonVisibleArray: any[] = []; // Array llenado cuando se activa cada método de la extracción de información
  buttonVisible: boolean = false; // Visibilidad del botón para exportar a Excel. Dependiente de buttonVisibleArray
  filterVisible: boolean = false; // Visibilidad del filtro.

  /**
   * Maneja la visibilidad del botón para exportar los datos a Excel.
   * @method showButton
   */
  showButton() {
    // Cada vez que se llama agrega un elemto a buttonVisibleArray
    this.buttonVisibleArray.push("x");
    // Cuando se extraigan todos los datos (3 tablas) haz visible el botón.
    if(this.buttonVisibleArray.length === 3) {
      this.buttonVisible = true;
    }
  }

  /**
   * Hace visible el filtro.
   * @method showFilter
   */
  showFilter() {
    this.filterVisible = true
  }

  /**
   * Esconde el filtro y elimina los valores de fecha1 y fecha2.
   * @method closeFilter
   */
  closeFilter() {
    this.filterVisible = false;
    this.fecha1 = "";
    this.fecha2 = "";
  }

  /**
   * Extrae los datos de la tabla `formato_auditoria`.
   * @method getAuditsToExcel
   * @returns {Subscription} Devuelve una suscripción a los datos de `formato_auditoria`
   */
  getAuditsToExcel(): Subscription {
    return this.auditsService.getAllAudits().subscribe({
      // si es éxitosa asigna un valor a `auditList` y llama al método `showButton`
      next: (data) => {
        this.auditsList = data;
        this.showButton()
      },
      // Si falla muestra el error en consola.
      error: (e) => console.error(e)
    })
  }

  /**
   * Extrae los datos de la tabla `auditoria_del_proceso`.
   * @method getProcessToExcel
   * @returns {Subscription} Devuelve una suscripción a los datos de `auditoria_del_proceso`
   */
  getProcessToExcel(): Subscription {
    return this.processService.getAllProcesses().subscribe({
      // Si es éxitosa asigna un valor a `processList` y llama a `getPDToExcel`
      next: (data) => {
        this.processList = data;
        this.getPDToExcel();
      },
      // Si falla imprime el error en consola
      error: (e) => console.error(e)
    })
  }

  /**
   * Extrae los datos de la tabla `pedidos`.
   * @method getOrdersToExcel
   * @returns {Subscription} Devuelve una suscripción a los datos de `pedidos`
   */
  getOrdersToExcel(): Subscription {
    return this.ordersService.getAllOrders().subscribe({
      // Si es éxitosa asigna un valor a `ordersList` y llama a `getOIToExcel`
      next: (data) => {
        this.ordersList = data;
        this.getOIToExcel();
      },
      // Si hay un error, muestralo por consola
      error: (e) => console.error(e)
    })
  }

  /**
   * Extrae los datos de la tabla `proceso_defecto`.
   * @method getPDToExcel
   * @returns {Subscription} Devuelve una suscripción a los datos de `proceso_defecto`
   */
  getPDToExcel(): void {
    this.pdService.getAllPD().subscribe({
      // Si es éxitosa asigna un valor a `pdList` y llama a `showButton`
      next: (data) => {
        this.pdList = data;
        this.showButton()
      },
      // Si falla, muestra el error en consola.
      error: (e) => console.error(e)
    })
  }

  /**
   * Extrae los datos de la tabla `pedido_item`.
   * @method getOIToExcel
   * @returns {Subscription} Devuelve una suscripción a los datos de `pedido_item`
   */
  getOIToExcel(): void {
    this.oiService.getAllOI().subscribe({
      // Si es éxitosa asigna un valor a `oiList` y llama a `showButton`
      next: (data) => {
        this.oiList = data;
        this.showButton()
      },
      // Si falla, muestra el error en consola
      error: (e) => console.error(e)
    })
  }

  /**
   * Establece la información según los parámetros establecidos en el filtro.
   * @method
   * @returns {any} Es un método void, pero tiene un return para que ya no ejecute codigo innecesario
   */
  exportExcel(): void {
    // Asigna los valores de la información obtenida en constantes
    const audits = this.dataOfAudits();
    const process = this.dataOfProcess();
    const orders = this.dataOfOrders();

    // variable que almacenara la información final
    let data: any;

    // Si no hay parámetros en el filtro, exporta toda la información existente en las tablas.
    if(this.fecha1 === "" && this.fecha2 === "") {
      data = []
      
      // Solo exporta las tablas que contengan información.
      audits.length > 0 ? data.push({data: audits, sheetName: "Formato Auditoria"}) : "";
      // Vacía los campos repetidos de las tablas, que tiene multiples registros es sus tablas unión
      process.length > 0 ? data.push({data: this.nullFields(process, ["Defecto"], "#PNC") , sheetName: "Auditoria del proceso"}) : "";
      orders.length > 0 ? data.push({data: this.nullFields(orders, ["Código item", "Cantidad", "Tecnologia item"], "Código pedido"), sheetName: "Pedidos"}) : "";
      
      // Oculta el filtro
      this.filterVisible = false;
      // Termina el método activando el método `exportAsExcel` del servicio `excelService`
      return this.excelService.exportAsExcel(data)
    }

    // Si hay algún parámetro en el filtro, obten la información que lo cumpla
    data = this.convertData(audits, process, orders);
    // Si hay información que cumpla con el parámetro, conviertela a Excel, sino solo oculta el filtro
    data ? this.excelService.exportAsExcel(data) : this.filterVisible = false;

  }

  /**
   * Filtra la información formateada si hay parámetros en el filtro.
   * @method
   * @param {any} audits Información de `formato_auditoria` formateada.
   * @param {any} process Información de `auditoria_del_proceso` y `proceso_defecto` formateada.
   * @param {any} orders Información de `pedidos` y `pedido_item` fomateada.
   * @returns {any} Si hay información que cumpla con los parámetros, devuelvela como un array de objetos
   *  y si no, solo regresa para terminar el método
   */
  convertData(audits: any, process: any, orders:any) {
    // Filtra toda la información formateada, con los parámetros establecidos
    const auditsFiltered = this.filterService.filterData(audits, { fecha1: this.fecha1, fecha2: this.fecha2 })
    const processFiltered = this.filterService.filterData(process, { fecha1: this.fecha1, fecha2: this.fecha2 })
    const ordersFiltered = this.filterService.filterData(orders, { fecha1: this.fecha1, fecha2: this.fecha2 })

    // Crea una constante que almacenara la información final
    const data = [];

    // VVacía los campos repetidos de las tablas, que tiene multiples registros es sus tablas unión
    const processFilteredAndNull = this.nullFields(processFiltered, ["Defecto"], "#PNC")
    const ordersFilteredAndNull = this.nullFields(ordersFiltered, ["Código item", "Cantidad", "Tecnologia item"], "codigoPedido")
    
    // Solo añade la tabla a la información final si tiene registros
    auditsFiltered.length > 0 ? data.push({data: auditsFiltered, sheetName: "Formato Auditoria"}) : "",
    processFiltered.length > 0 ? data.push({data: processFilteredAndNull, sheetName: "Auditoria del proceso"}) : "",
    ordersFiltered.length > 0 ? data.push({data: ordersFilteredAndNull, sheetName: "Pedidos"}) : ""

    // Lanza una alerta de error y regresa si ninguna tabla cumple con los 
    if(data.length === 0) {
      this.alertsService.errorAlert();
      return;
    }

    return data;
  }

  /**
   * Formatea los campos de `auditsList`
   * @method
   * @returns {Array<any>} Retorna un array de objetos con la información formateada
   */
  dataOfAudits() {
    // Almacena la información resultante en la constante `data`
    const data = this.auditsList ? 
    // Si existe `auditList` crea un objeto con la siguente información por cada registro existente.
      this.auditsList.map(audit => ({
        PO: audit.PO,
        fecha_auditoria: audit.fecha_auditoria,
        "Código Pedido": audit.codigoPedido,
        "Código Item": audit.codigoItem,
        "Hora inicio": audit.hora_inicio,
        "Hora fin": audit.hora_fin,
        Tiempo: audit.tiempo,
        Auditor: audit.usuario.nombre,
        "Tipo de auditoria": audit.tipo_auditoria,
        Tecnologia: audit.tecnologia.nombre,
        Alertas: audit.alertas_proceso,
        "Cantidad del lote": audit.cantidad_lote,
        "Tipo de muestra": audit.tipo_muestra,
        AQL: audit.tipo_muestra === "reducida" ? 
          audit.aql.muestra_inspeccion_reducida : audit.tipo_muestra === "normal" ? 
          audit.aql.muestra_inspeccion_normal : audit.aql.muestra_inspeccion_severa,
        "Status": audit.estado
      })) : []
    return data;
  }

  /**
   * Formatea los campos de `auditoria_del_proceso` y `proceso_defecto`. Crea un objeto por cada 
   * registro de `proceso_defecto` la cual puede tener multiples registros para un solo registro de 
   * `auditoria_del_proceso`
   * @method
   * @returns {Array<any>} Retorna un array de objetos con los capos formateados.
   */
  dataOfProcess() {
    // Allmacena la información reusltante en la constante `data`
    const data = this.processList && this.pdList ? 
    // Si existe `processList` y `pdList` itera por cada objeto de `processList`
      this.processList.map(process => {
        // Encuentra los objetos en `pdList` donde la propiedad `IDproceso` sea la misma que en `processList`
        const pdFiltered = this.pdList?.filter(pd => pd.IDproceso === process.IDproceso)
        // Retorna un objeto por cada objeto de `pdList`
        return pdFiltered?.map(pd => ({
          "#PNC": process.pnc,
          fecha_apertura: process.fecha_apertura,
          "Cantidad rechazada": process.cantidad_rechazada,
          "Defectos encontrados": process.defectos_encontrados,
          Defecto: pd.codigoDefecto,
          Operario: process.operario,
          "Acción": process.accion,
          "Aprobado por": process.aprobado_por,
          "Comentario de la acción": process.comentario_accion,
          "Cantidad de descarte": process.cantidad_descarte,
          "Cantidad aceptada": process.cantidad_aceptada,
          "Observación": process.observacion,
          fecha_cierre: process.fecha_cierre,
          "Re auditoria": process.re_auditoria,
          "Estado re-auditoria": process.estado_re_auditoria,
          "PO auditoria": process.formato_auditorium.PO
        }))
      }) : []

    // Convierte el array de arrays reulstante a un array de objetos y retornalo.
    return data.flat();
  }

  /**
   * Formatea los campos de `pedidos` y `pedido_item`. Crea un objeto por cada registro de 
   * `proceso_defecto` la cual puede tener multiples registros para un solo registro de 
   * `pedidos`
   * @method
   * @returns {Array<any>} Retorna un array de objetos con los campos formateados.
   */
  dataOfOrders() {
    // Almacena el resultado en la constante `data`
    const data = this.ordersList && this.oiList
      // Si existe `ordersList` y `oiList`, itera por cada elemento de `ordersList`
        ? this.ordersList.map(order => {
          // Encuentra los elemtos de `oiList` donde su propiedad `codigoPedido` sea igual en `ordersList`
            const oiFiltered = this.oiList?.filter(oi => oi.codigoPedido === order.codigoPedido);
            // Retorna un objeto por cada elemento de `oiList`
            return oiFiltered?.map(oi => ({
                "Código pedido": order.codigoPedido,
                "Descripción": order.descripcion,
                Tecnologia: order.tecnologia.nombre,
                "Código cliente": order.codigoCliente,
                "Código item": oi.item.codigoItem,
                Cantidad: oi.cantidad,
                "Tecnologia item": oi.item.tecnologia.nombre,
                createdAt: order.createdAt
            }));
        })
        : [];

    // Convierte el array de arrays resultante en un array de objetos.
    return data.flat();
}

/**
 * Vacía los campos repetidos de la información formateada, donde los campos de sus tablas unión 
 * pertenezca a un mismo registro de una de las tablas principales.
 * @method
 * @param {any[]} array 
 * @param {String[]} immutableFields 
 * @param {String} id 
 * @returns {any[]} Retorna el array recibido pero con los campos repetidos de cada objeto vacíos.
 */
  nullFields(array: any[], immutableFields: string[], id: string) {
    // Regresa si no hay un array u este no tiene información
    if(!array || array.length === 0) return;
    // Declara el identificador actual
    let currentIdentifier;
    // Declara el primer objeto (de los registros que tenga el mismo valor de `id`)
    let firstObj;
    // Itera por cada objeto del array
    for (const obj of array) {
      // Si el valor de `id` cambia, cambia el valor de `currentIdentifier` y `firstObject`
      if (currentIdentifier !== obj[id]) {
        currentIdentifier = obj[id]
        firstObj = obj
      }
      // Itera por cada pripoedad del objeto
      for (const field in obj) {
        // Vacía los campos si estos no son elemento de `immutableFields` o si no es el primer objeto
          if (!immutableFields.includes(field) && obj !== firstObj) {
            obj[field] = null;
          } 
      }
    }
    // Regresa el array modificado
    return array;
  }
}
