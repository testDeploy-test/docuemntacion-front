import { Component, ElementRef, OnInit, Renderer2 } from '@angular/core';
import { Title } from '@angular/platform-browser';
import { PedidosService } from '../../../services/pedidos.service';
import { PedidosModel } from '../../../models/pedidos.model';
import { NoDataComponent } from '../../no-data/no-data.component';
import { PedidosFormComponent } from '../../forms/pedidos-form/pedidos-form.component';
import { AlertsService } from '../../../services/alerts.service';
import { PedidoItemModel } from '../../../models/pedido-item.model';
import { PedidoItemService } from '../../../services/pedido-item.service';
import { SearchComponent } from '../../search/search.component';

/**
 * Componente de la tabla `pedidos`
 * @implements {OnInit}
 */
@Component({
  selector: 'app-pedidos',
  standalone: true,
  imports: [PedidosFormComponent, NoDataComponent, SearchComponent],
  templateUrl: './pedidos.component.html',
  styleUrl: './pedidos.component.css'
})
export class PedidosComponent implements OnInit {

  ordersList?: PedidosModel[]; // Declara la lista de registros de `pedidos`
  piList?: PedidoItemModel[]; // Declara la lista de registros de `pedido_item`

  ordersFormVisible: boolean = false; // Visibilidad del formulario
  action: string = ""; // Acción del formulario ("create" o "update") 
  selectedOrder: any; // Registro seleccionado para actualizar

  listLength: number = 0; // Cantidad de elementos totales de la tabla `pedidos`
  table: string = "pedidos" // Nombre de la tabla (ruta) para el componente search

  /**
   * @constructor
   * @param {Title} titleService Servicio del titulo
   * @param {PedidosService} ordersService Servicio de la tabla `pedidos`
   * @param {PedidoItemService} piService Servicio de la tabla `pedido_item`
   * @param {AlertsService} alertsService Servicio de alertas
   * @param {ElementRef} el Servicio para acceder a elementos del HTML
   * @param {Renderer2} renderer Servicio para la renderización del DOM
   */
  constructor(
    private titleService: Title, 
    private ordersService: PedidosService,
    private piService: PedidoItemService,
    private alertsService: AlertsService,
    private el: ElementRef,
    private renderer: Renderer2,
  ) {}

  /**
   * Al inicializar el componente cambia el titulo y recupera los registros de la tabla `pedidos` y
   * `pedido_item` a través de la llamada a los métodos correspondientes
   */
  ngOnInit(): void {
    this.titleService.setTitle("Pedidos");
    this.getAll();
    this.getPI();
  }

  /**
   * Método para obtener todos los registros de la tabla `pedido_item`.Si es éxitoso
   * asigna los registros a `piList` y si falla muestra el error en consola
   * @method
   */
  public getPI(): void {
    this.piService.getAllOI().subscribe({
      next: (data) => {
        this.piList = data;
      },
      error: (e) => console.error(e)
    })
  }

  /**
   * Método para recuperar todos los registros de la tabla `pedidos`. Si es éxitoso asigna
   * los registros a `ordersList` y si falla muestra el error en la consola
   * @method
   */
  public getAll(): void {
    this.ordersService.getAllOrders().subscribe({
      next: (data) => {
        this.ordersList = data;
        this.listLength = data.length;
      },
      error: (e) => console.error(e)
    })
  }

  /**
   * Método para eliminar un registro de `pedidos` en base a `codigoPedido`.
   * Pide confirmación de la acción. En caso de confirmar y si la acción es éxitosa muestra una
   * alerta de éxito y recarga los registros, si falla muestra el errror en la consola y muestra una
   * alerta de error
   * @method
   * @param {any} codigoPedido Código del registro a eliminar
   */
  public delete(codigoPedido: any): void {
    this.alertsService.confirmAlert("You are about to delete a record <br> It is not recommended.", "delete").then((confirmed) => {
      if(confirmed) {
        this.ordersService.deleteOrder(codigoPedido).subscribe({
          next: () => {
            this.alertsService.succesAlert(`The record ${codigoPedido} deleted.`);
            this.getAll();
          },
          error: (e) => {
            console.error(e);
            this.alertsService.errorAlert();
          }
        })
      }
    })
  }

  /**
   * Método para visualizar el formulario con la acción de creación de un nuevo registro 
   * @method
   */
  public showOrderFormCreate(): void {
    // Muestra el formulario
    this.ordersFormVisible = true;
    // Especifica la acción
    this.action = "create";
    // Llama al método `disabled`
    this.disabled();
  }

  /**
   * Método para visualizar el formulario con la acción de actualización del registro
   * @method
   * @param {any} order Toda la información del registro a atualizar
   */
  public showOrderFormUpdate(order: any): void {
    // Muestra el formulario
    this.ordersFormVisible = true;
    // Asigna el valor de `selectedOrder`
    this.selectedOrder = order;
    // Especifica la acción
    this.action = "update";
    // Llama al método `disabled`
    this.disabled();
  }

  /**
   * Método para deshabilitar todas las funciones de creación, actualización y 
   * eliminación.
   * @method
   * @returns {void} Retorna si no hay registros en la tabla `pedidos`
   */
  public disabled(): void {
    if(!this.ordersList || this.ordersList.length === 0) return;
    const disabled = this.el.nativeElement.querySelector(".table-container");
    const button = this.el.nativeElement.querySelector(".form-create-button");

    this.renderer.addClass(disabled, "disabled");
    this.renderer.addClass(button, "disabled");
  }

    /**
   * Método para habilitar todas las funciones de creación, actualización y 
   * eliminación.
   * @method
   * @returns {void} Retorna si no hay registros en la tabla `pedidos`
   */
  public enabled(): void {
    if(!this.ordersList || this.ordersList.length === 0) return;
    const disabled = this.el.nativeElement.querySelector(".table-container");
    const button = this.el.nativeElement.querySelector(".form-create-button");

    this.renderer.removeClass(disabled, "disabled");
    this.renderer.removeClass(button, "disabled");
  }
}

                            }

                            @for(pi of piList; track pi.IDpedido_item) {
                                @if(pi.codigoPedido === order.codigoPedido) {

                            <div class="actions-container">
                                <button class="update-button" (click)="showOrderFormUpdate(order)">Editar</button>
                                <button class="delete-button" (click)="delete(order.codigoPedido)">Eliminar</button>
                            </div>
                        </td>
                    </tr>
                }
            </tbody>
        </table>
    </div>
}

@if(ordersFormVisible) {
    <app-pedidos-form [action]="action" [selectedOrder]="selectedOrder" (enabledTable)="enabled()" (getAll)="getAll()" (getPI)="getPI()" (hiddenForm)="ordersFormVisible = $event"></app-pedidos-form>
}
