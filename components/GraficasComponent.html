import { Component, OnInit } from '@angular/core';
import { NavbarComponent } from '../../navbar/navbar.component';
import { FooterComponent } from '../../footer/footer.component';
import { Title } from '@angular/platform-browser';
import { FormsModule } from '@angular/forms';
import { FilterService } from '../../../services/filter.service';
import {DefectGraphicsComponent} from "../defect-graphics/defect-graphics.component"
import { AqlGraphicsComponent } from '../aql-graphics/aql-graphics.component';
import { UserGraphicsComponent } from '../users/user-graphics/user-graphics.component';
import { CustomersGraphicsComponent } from '../customers-graphics/customers-graphics.component';
import { OrderItemsGraphicsComponent } from '../items/order-items-graphics/order-items-graphics.component';
import { ItemsDefectsGraphicsComponent } from '../items/items-defects-graphics/items-defects-graphics.component';
import { provideIcons, NgIconComponent } from '@ng-icons/core';
import { bootstrapFilter, bootstrapX } from "@ng-icons/bootstrap-icons";
import { UserTimeComponent } from '../users/user-time/user-time.component';

/**
 * Componete de las gráficas
 * @implements {OnInit}
 */
@Component({
  selector: 'app-graficas',
  standalone: true,
  imports: [FormsModule, NgIconComponent, NavbarComponent, FooterComponent, DefectGraphicsComponent, 
    AqlGraphicsComponent, UserGraphicsComponent, CustomersGraphicsComponent, OrderItemsGraphicsComponent,
    ItemsDefectsGraphicsComponent, UserTimeComponent
  ],
  viewProviders: [provideIcons({bootstrapFilter, bootstrapX})],
  templateUrl: './graficas.component.html',
  styleUrl: './graficas.component.css'
})
export class GraficasComponent implements OnInit {

  /**
   * Al inicializar el componente cambia el titulo
   */
  ngOnInit(): void {
    this.titleService.setTitle("Gráficas")
  }

  /**
   * @constructor
   * @param {Title} titleService 
   * @param {FilterService} filterService 
   */
  constructor(private titleService: Title, private filterService: FilterService) {}


  fecha1: string = ""; // Fecha de inicio para el filtro 
  fecha2: string = ""; // Fecha de finalización para el filtro

  filterVisible: boolean = false; // Visibilidad del filtro

  // Visibilidad de los coponentes de cada gráfica
  showDefectsChart: boolean = true;
  showAqlChart: boolean = true;
  showCustomersChart: boolean = true;
  showUserChart: boolean = true;
  showUserTimeChart: boolean = true;
  showOrderItemsChart: boolean = true;
  showItemDefectsChart: boolean = true;
  
  /**
   * Método para visualizar el filtro
   * @method
   */
  showFilter() {
    this.filterVisible = true
  }

  /**
   * Método para ocultar el filtro
   * @method
   */
  closeFilter() {
    this.filterVisible = false;
  }

  /**
   * Método para activar las funciones de filtro de todos los componente hijos
   * con los parámetros `fecha1` y `fecha2`. Una vez ejecutado oculta el filtro y
   * vacía los parámetros
   * @method
   */
  filtrar() {
    this.filterService.filter()
    this.filterVisible = false;
    this.fecha1 = "";
    this.fecha2 = "";
  }
}
