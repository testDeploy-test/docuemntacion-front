import { Component, OnInit, Input, Output, EventEmitter } from '@angular/core';
import { Chart, ChartType } from "chart.js/auto";
import { PedidosModel } from '../../../models/pedidos.model';
import { PedidosService } from '../../../services/pedidos.service';
import { Subscription } from 'rxjs';
import { FilterService } from '../../../services/filter.service';
import { ColorsService } from '../../../services/colors.service';

/**
 * Componente de la relación Pedido-Cliente
 * @implements {OnInit}
 */
@Component({
  selector: 'app-customers-graphics',
  standalone: true,
  imports: [],
  templateUrl: './customers-graphics.component.html',
  styleUrl: './customers-graphics.component.css'
})
export class CustomersGraphicsComponent implements OnInit {

  /**
   * Al inicializar el componente extrae los registros de la tabla `pedidos` y
   * subscribe a éste el método `createCustomerData`
   * @method
   */
  ngOnInit(): void {
    this.getCustomersOfOrders().add(() => {
      this.createCustomerData();
    })
  }

  /**
   * @constructor
   * @param {PedidosService} ordersService Servicio de la tabla `pedidos`
   * @param {FilterService} filterService Servicio del filtro
   * @param {ColorsService} colorService Servicio para la asignación de colores
   */
  constructor(
    private ordersService: PedidosService, 
    private filterService: FilterService,
    private colorService: ColorsService
  ) {
    this.filterService.$emitter.subscribe(() => {
      this.filter();
    })
  }

  @Input() fecha1: string = ""; // Parámetro enviado desde el padre. Utilizado para el filtrado (inicio)
  @Input() fecha2: string = ""; // Parámetro enviado desde el padre. Utilizado para el filtrado (final)

  @Output() chartVisible = new EventEmitter<boolean> // Evento para cambiar la visibilidad del componente

  chart: any; // Declaración del gráfico
  list?: PedidosModel[]; // Declara la lista de registros de `pedidos` (modificable)
  listOriginal? : PedidosModel[]; // declara la lista de registros de `pedidos`

  customersList: string[] = []; // Declara la lista de los clientes
  repetitionsList: number[] = []; // Declara la lista de la cantidad de pedidos realizados por cada cliente

  /**
   * Método para obtener los registros de la tabla `pedidos` la cual es la tabla necesaria
   * para este gráfico
   * @method
   * @returns {Subscription} Subscripción a los resultados de éste método para su futura utilización por
   * el método `createCustomerData`
   */
  getCustomersOfOrders(): Subscription {
    return this.ordersService.getAllOrders().subscribe({
      next: (data) => {
        this.list = data;
        this.listOriginal = data;
      },
      error: (e) => console.error(e)
    })
  }

  /**
   * Método para generar la información necesaria para el gráfico.
   * Si `list` existe almacena los clientes en `customersList` y la cantidad total de
   * pedidos que han realizados en `repetitionsList`
   * @method
   * @returns {void} Si `list` existe regresa una vez completada toda la operación para no ocultar el 
   * gráfico
   */
  createCustomerData(): void {
    const customerRepetitions = new Map<string, number>();

    if(this.list) {
      for(const info of this.list) {
        const customer = info.codigoCliente;
        const insert = customer ? customer : ""
        if(customerRepetitions.has(insert)) {
          customerRepetitions.set(insert, customerRepetitions.get(insert)! + 1);
        } else {
          customerRepetitions.set(insert, 1)
        }
      }

      const array = Array.from(customerRepetitions).sort((a, b) => b[1] - a[1])

      array.forEach((value) => {
        this.customersList.push(value[0]);
        this.repetitionsList.push(value[1])
      })
      this.createCustomerChart();
      return;
    }
    this.chartVisible.emit(false);
  }

  /**
   * Método para la creación del gráfico correspondiente.
   * Se especifica el tipo, y la información correspondiente extraida con el método `createCustomerData` además
   * de que se establecen unas configuraciónes del gráfico
   * @method
   */
  createCustomerChart() {
    this.chart = new Chart("CustomersChart", {
      type: "bar" as ChartType,
      data: {
        labels: this.customersList,
        datasets: [{
          data: this.repetitionsList,
          backgroundColor: this.colorService.getColors(this.customersList.length)
        }]
      },
      options: {
        responsive: true,
        plugins: {
          title: { display: true, text: "Relación Pedidos-Clientes" },
          legend: { display: false }
        },
        scales: {
          x: {
            title: { display: true, text: "Código Cliente" }
          },
          y: {
            ticks: { stepSize: 1 },
            title: { display: true, text: "Pedidos realizados" }
          }
        },
        maintainAspectRatio: false
      }
    })
  }

  /**
   * Método para filtrar los registros por las fechas proporcionadas por el componente padre
   * `fecha1` y `fecha2`
   * @method
   * @returns {void} Regresa si no hay registros coincidentes con los parámetros establecidos u retorna
   * la ejecución del método `getCustomersOfOrders` y subscribe a esto el método `createCustomerData` si los parámetros están
   * vacíos
   */  
  filter() {
    // Destruye el gráfico actual
    this.chart.destroy();
    // Vacía las listas de los datos
    this.repetitionsList = [];
    this.customersList = [];
    // haz bisible el gráfico
    this.chartVisible.emit(true)
    // Si los parámetros de filtrado están vacíos retorna la ejecución de `getCustomersOfOrders` subscribiendo a esto a `createCustomerData` 
    if(this.fecha1 === "" && this.fecha2 === "") {
      return this.getCustomersOfOrders().add(() => {
        this.createCustomerData();
      })
    }

    // Filtra los registros según los parámetros
    const data = this.filterService.filterData(this.listOriginal, { fecha1: this.fecha1, fecha2: this.fecha2 })
    // Si no hay registros que cumplan con los parámetros, oculata el gráfico y regresa
    if(data.length === 0) {
      this.chartVisible.emit(false);
      return;
    }
    
    // Si existen registros que cumplan los parámetros asignalos a `list` y vuleve a generar la información pertinente
    this.list = data;
    this.createCustomerData();
  }
}
